#191 NUMBER OF 1 BITS

Count the number of set bits (Hamming weight).

public int hammingWeight(int n) {
    int count = 0;
    while (n != 0) {
        count += n & 1;
        n >>>= 1;
    }
    return count;
}

Complexity: O(32) Time | O(1) Space

#338 COUNTING BITS

Return an array of the number of 1's in binary representation of every number 0 to n.

public int[] countBits(int n) {
    int[] dp = new int[n + 1];
    int offset = 1;
    for (int i = 1; i <= n; i++) {
        if (offset * 2 == i) offset = i;
        dp[i] = 1 + dp[i - offset];
    }
    return dp;
}

Complexity: O(n) Time | O(n) Space

#190 REVERSE BITS

Reverse bits of a given 32-bit unsigned integer.

public int reverseBits(int n) {
    int result = 0;
    for (int i = 0; i < 32; i++) {
        result = (result << 1) | (n & 1);
        n >>>= 1;
    }
    return result;
}

Complexity: O(32) Time | O(1) Space

#268 MISSING NUMBER

Find the missing number in range [0, n] using XOR.

public int missingNumber(int[] nums) {
    int res = nums.length;
    for (int i = 0; i < nums.length; i++) {
        res ^= i ^ nums[i];
    }
    return res;
}

Complexity: O(n) Time | O(1) Space

#371 SUM OF TWO INTEGERS

Calculate the sum of two integers without using + or -.

public int getSum(int a, int b) {
    while (b != 0) {
        int carry = (a & b) << 1;
        a = a ^ b;
        b = carry;
    }
    return a;
}

Complexity: O(32) Time | O(1) Space

#53 MAXIMUM SUBARRAY

Find the contiguous subarray with the largest sum (Kadane's Algorithm).

public int maxSubArray(int[] nums) {
    int maxSum = nums[0], curSum = 0;
    for (int n : nums) {
        curSum = Math.max(n, curSum + n);
        maxSum = Math.max(maxSum, curSum);
    }
    return maxSum;
}

Complexity: O(n) Time | O(1) Space

#55 JUMP GAME

Determine if you can reach the last index.

public boolean canJump(int[] nums) {
    int goal = nums.length - 1;
    for (int i = nums.length - 2; i >= 0; i--) {
        if (i + nums[i] >= goal) goal = i;
    }
    return goal == 0;
}

Complexity: O(n) Time | O(1) Space

#45 JUMP GAME II

Minimum number of jumps to reach the last index.

public int jump(int[] nums) {
    int jumps = 0, curEnd = 0, farthest = 0;
    for (int i = 0; i < nums.length - 1; i++) {
        farthest = Math.max(farthest, i + nums[i]);
        if (i == curEnd) {
            jumps++;
            curEnd = farthest;
        }
    }
    return jumps;
}

Complexity: O(n) Time | O(1) Space

#134 GAS STATION

Find the starting gas station index to complete the circuit.

public int canCompleteCircuit(int[] gas, int[] cost) {
    int totalSurplus = 0, curSurplus = 0, start = 0;
    for (int i = 0; i < gas.length; i++) {
        totalSurplus += gas[i] - cost[i];
        curSurplus += gas[i] - cost[i];
        if (curSurplus < 0) {
            curSurplus = 0;
            start = i + 1;
        }
    }
    return totalSurplus >= 0 ? start : -1;
}

Complexity: O(n) Time | O(1) Space

#846 HAND OF STRAIGHTS

Check if a hand can be rearranged into groups of consecutive cards.

public boolean isNStraightHand(int[] hand, int groupSize) {
    if (hand.length % groupSize != 0) return false;
    TreeMap<Integer, Integer> count = new TreeMap<>();
    for (int card : hand)
        count.merge(card, 1, Integer::sum);

    while (!count.isEmpty()) {
        int first = count.firstKey();
        for (int i = first; i < first + groupSize; i++) {
            if (!count.containsKey(i)) return false;
            count.merge(i, -1, Integer::sum);
            if (count.get(i) == 0) count.remove(i);
        }
    }
    return true;
}

Complexity: O(n log n) Time | O(n) Space

#355 DESIGN TWITTER

Design a simplified version of Twitter.

class Twitter {
    private int timestamp = 0;
    private Map<Integer, Set<Integer>> followMap = new HashMap<>();
    private Map<Integer, List<int[]>> tweetMap = new HashMap<>();

    public void postTweet(int userId, int tweetId) {
        tweetMap.computeIfAbsent(userId, k -> new ArrayList<>())
                .add(new int[]{timestamp++, tweetId});
    }

    public List<Integer> getNewsFeed(int userId) {
        PriorityQueue<int[]> pq = new PriorityQueue<>(
            (a, b) -> b[0] - a[0]);
        followMap.computeIfAbsent(userId, k -> new HashSet<>()).add(userId);
        for (int followee : followMap.get(userId)) {
            List<int[]> tweets = tweetMap.get(followee);
            if (tweets != null) {
                int idx = tweets.size() - 1;
                int[] t = tweets.get(idx);
                pq.offer(new int[]{t[0], t[1], followee, idx});
            }
        }
        List<Integer> feed = new ArrayList<>();
        while (!pq.isEmpty() && feed.size() < 10) {
            int[] curr = pq.poll();
            feed.add(curr[1]);
            int idx = curr[3] - 1;
            if (idx >= 0) {
                int[] t = tweetMap.get(curr[2]).get(idx);
                pq.offer(new int[]{t[0], t[1], curr[2], idx});
            }
        }
        return feed;
    }

    public void follow(int followerId, int followeeId) {
        followMap.computeIfAbsent(followerId, k -> new HashSet<>())
                 .add(followeeId);
    }

    public void unfollow(int followerId, int followeeId) {
        Set<Integer> set = followMap.get(followerId);
        if (set != null) set.remove(followeeId);
    }
}

Complexity: O(k log n) getNewsFeed | O(1) other ops

#380 INSERT DELETE GETRANDOM O(1)

Design a set structure with insert, remove, and getRandom all in O(1).

class RandomizedSet {
    private List<Integer> list = new ArrayList<>();
    private Map<Integer, Integer> map = new HashMap<>();
    private Random rand = new Random();

    public boolean insert(int val) {
        if (map.containsKey(val)) return false;
        map.put(val, list.size());
        list.add(val);
        return true;
    }

    public boolean remove(int val) {
        if (!map.containsKey(val)) return false;
        int idx = map.get(val);
        int last = list.get(list.size() - 1);
        list.set(idx, last);
        map.put(last, idx);
        list.remove(list.size() - 1);
        map.remove(val);
        return true;
    }

    public int getRandom() {
        return list.get(rand.nextInt(list.size()));
    }
}

Complexity: O(1) all operations | O(n) Space

#7 REVERSE INTEGER

Reverse digits of a 32-bit signed integer with overflow detection.

public int reverse(int x) {
    int result = 0;
    while (x != 0) {
        int digit = x % 10;
        x /= 10;
        if (result > Integer.MAX_VALUE / 10 ||
            result < Integer.MIN_VALUE / 10)
            return 0;
        result = result * 10 + digit;
    }
    return result;
}

Complexity: O(log x) Time | O(1) Space

#8 STRING TO INTEGER (ATOI)

Implement atoi which converts a string to an integer with edge case handling.

public int myAtoi(String s) {
    int i = 0, sign = 1, n = s.length();
    long result = 0;
    while (i < n && s.charAt(i) == ' ') i++;
    if (i < n && (s.charAt(i) == '+' || s.charAt(i) == '-')) {
        sign = (s.charAt(i) == '-') ? -1 : 1;
        i++;
    }
    while (i < n && Character.isDigit(s.charAt(i))) {
        result = result * 10 + (s.charAt(i) - '0');
        if (result * sign >= Integer.MAX_VALUE) return Integer.MAX_VALUE;
        if (result * sign <= Integer.MIN_VALUE) return Integer.MIN_VALUE;
        i++;
    }
    return (int)(result * sign);
}

Complexity: O(n) Time | O(1) Space
