#125 VALID PALINDROME

Check if a string is a palindrome, considering only alphanumeric characters.

public boolean isPalindrome(String s) {
    int l = 0, r = s.length() - 1;
    while (l < r) {
        while (l < r && !Character.isLetterOrDigit(s.charAt(l))) l++;
        while (r > l && !Character.isLetterOrDigit(s.charAt(r))) r--;
        if (Character.toLowerCase(s.charAt(l)) !=
            Character.toLowerCase(s.charAt(r))) return false;
        l++; r--;
    }
    return true;
}

Complexity: O(n) Time | O(1) Space

#15 3SUM

Find all unique triplets that sum to zero.

public List<List<Integer>> threeSum(int[] nums) {
    List<List<Integer>> res = new ArrayList<>();
    Arrays.sort(nums);
    for (int i = 0; i < nums.length; i++) {
        if (i > 0 && nums[i] == nums[i - 1]) continue;
        int l = i + 1, r = nums.length - 1;
        while (l < r) {
            int sum = nums[i] + nums[l] + nums[r];
            if (sum > 0) r--;
            else if (sum < 0) l++;
            else {
                res.add(Arrays.asList(nums[i], nums[l], nums[r]));
                l++;
                while (l < r && nums[l] == nums[l - 1]) l++;
            }
        }
    }
    return res;
}

Complexity: O(n²) Time | O(1) Space

#11 CONTAINER WITH MOST WATER

Find two lines that form a container holding the most water.

public int maxArea(int[] height) {
    int l = 0, r = height.length - 1, res = 0;
    while (l < r) {
        res = Math.max(res,
            Math.min(height[l], height[r]) * (r - l));
        if (height[l] < height[r]) l++;
        else r--;
    }
    return res;
}

Complexity: O(n) Time | O(1) Space

#42 TRAPPING RAIN WATER

Calculate how much water can be trapped between bars after raining.

public int trap(int[] height) {
    if (height.length == 0) return 0;
    int l = 0, r = height.length - 1;
    int leftMax = height[l], rightMax = height[r];
    int res = 0;
    while (l < r) {
        if (leftMax < rightMax) {
            l++;
            leftMax = Math.max(leftMax, height[l]);
            res += leftMax - height[l];
        } else {
            r--;
            rightMax = Math.max(rightMax, height[r]);
            res += rightMax - height[r];
        }
    }
    return res;
}

Complexity: O(n) Time | O(1) Space

#167 TWO SUM II

Find two numbers in a sorted array that add up to a target.

public int[] twoSum(int[] numbers, int target) {
    int l = 0, r = numbers.length - 1;
    while (l < r) {
        int sum = numbers[l] + numbers[r];
        if (sum > target) r--;
        else if (sum < target) l++;
        else return new int[]{l + 1, r + 1};
    }
    return new int[]{};
}

Complexity: O(n) Time | O(1) Space

#283 MOVE ZEROES

Move all 0's to the end of an array while maintaining the relative order of non-zero elements.

public void moveZeroes(int[] nums) {
    int l = 0;
    for (int r = 0; r < nums.length; r++) {
        if (nums[r] != 0) {
            int temp = nums[l];
            nums[l] = nums[r];
            nums[r] = temp;
            l++;
        }
    }
}

Complexity: O(n) Time | O(1) Space

#121 BUY & SELL STOCK

Find the maximum profit you can achieve by buying and selling a stock once.

public int maxProfit(int[] prices) {
    int l = 0, maxP = 0;
    for (int r = 1; r < prices.length; r++) {
        if (prices[l] < prices[r]) {
            maxP = Math.max(maxP, prices[r] - prices[l]);
        } else {
            l = r;
        }
    }
    return maxP;
}

Complexity: O(n) Time | O(1) Space

#3 LONGEST SUBSTRING

Length of the longest substring without repeating characters.

public int lengthOfLongestSubstring(String s) {
    Set<Character> charSet = new HashSet<>();
    int l = 0, res = 0;
    for (int r = 0; r < s.length(); r++) {
        while (charSet.contains(s.charAt(r))) {
            charSet.remove(s.charAt(l));
            l++;
        }
        charSet.add(s.charAt(r));
        res = Math.max(res, r - l + 1);
    }
    return res;
}

Complexity: O(n) Time | O(n) Space

#424 REPEATING CHARACTER

Longest repeating character replacement with k operations.

public int characterReplacement(String s, int k) {
    int[] count = new int[26];
    int res = 0, l = 0, maxf = 0;
    for (int r = 0; r < s.length(); r++) {
        count[s.charAt(r) - 'A']++;
        maxf = Math.max(maxf, count[s.charAt(r) - 'A']);
        while ((r - l + 1) - maxf > k) {
            count[s.charAt(l) - 'A']--;
            l++;
        }
        res = Math.max(res, r - l + 1);
    }
    return res;
}

Complexity: O(n) Time | O(1) Space

#76 MINIMUM WINDOW SUBSTRING

Smallest substring containing all characters of another string.

public String minWindow(String s, String t) {
    if (t.isEmpty()) return "";
    Map<Character, Integer> countT = new HashMap<>();
    Map<Character, Integer> window = new HashMap<>();
    for (char c : t.toCharArray())
        countT.merge(c, 1, Integer::sum);

    int have = 0, need = countT.size();
    int[] res = {-1, -1};
    int resLen = Integer.MAX_VALUE, l = 0;

    for (int r = 0; r < s.length(); r++) {
        char c = s.charAt(r);
        window.merge(c, 1, Integer::sum);
        if (countT.containsKey(c) &&
            window.get(c).intValue() == countT.get(c).intValue())
            have++;
        while (have == need) {
            if ((r - l + 1) < resLen) {
                res = new int[]{l, r};
                resLen = r - l + 1;
            }
            char left = s.charAt(l);
            window.merge(left, -1, Integer::sum);
            if (countT.containsKey(left) &&
                window.get(left) < countT.get(left))
                have--;
            l++;
        }
    }
    return resLen == Integer.MAX_VALUE ? "" :
        s.substring(res[0], res[1] + 1);
}

Complexity: O(n) Time | O(n) Space

#239 SLIDING WINDOW MAX

Return the maximum element in each sliding window of size k.

public int[] maxSlidingWindow(int[] nums, int k) {
    int[] output = new int[nums.length - k + 1];
    Deque<Integer> deque = new ArrayDeque<>();
    int l = 0;
    for (int r = 0; r < nums.length; r++) {
        while (!deque.isEmpty() && nums[deque.peekLast()] < nums[r])
            deque.pollLast();
        deque.addLast(r);
        if (l > deque.peekFirst()) deque.pollFirst();
        if (r + 1 >= k) {
            output[l] = nums[deque.peekFirst()];
            l++;
        }
    }
    return output;
}

Complexity: O(n) Time | O(k) Space

#567 PERMUTATION IN STRING

Check if s1's permutation is a substring of s2.

public boolean checkInclusion(String s1, String s2) {
    if (s1.length() > s2.length()) return false;
    int[] s1Count = new int[26], s2Count = new int[26];
    for (int i = 0; i < s1.length(); i++) {
        s1Count[s1.charAt(i) - 'a']++;
        s2Count[s2.charAt(i) - 'a']++;
    }
    int matches = 0;
    for (int i = 0; i < 26; i++)
        if (s1Count[i] == s2Count[i]) matches++;

    int l = 0;
    for (int r = s1.length(); r < s2.length(); r++) {
        if (matches == 26) return true;
        int idx = s2.charAt(r) - 'a';
        s2Count[idx]++;
        if (s2Count[idx] == s1Count[idx]) matches++;
        else if (s2Count[idx] == s1Count[idx] + 1) matches--;

        idx = s2.charAt(l) - 'a';
        s2Count[idx]--;
        if (s2Count[idx] == s1Count[idx]) matches++;
        else if (s2Count[idx] == s1Count[idx] - 1) matches--;
        l++;
    }
    return matches == 26;
}

Complexity: O(n) Time | O(1) Space
