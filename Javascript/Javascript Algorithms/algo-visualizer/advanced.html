Segment Tree

O(log n)

Tree for efficient range queries and updates.

class SegmentTree {
    constructor(arr) {
        this.n = arr.length;
        this.tree = Array(4 * n).fill(0);
        this.build(arr, 0, 0, n - 1);
    }
    build(arr, node, start, end) {
        if (start === end) {
            this.tree[node] = arr[start];
        } else {
            const mid = (start + end) / 2;
            this.build(arr, 2*node+1, start, mid);
            this.build(arr, 2*node+2, mid+1, end);
            this.tree[node] = this.tree[2*node+1] + this.tree[2*node+2];
        }
    }
}

Fenwick Tree (BIT)

O(log n)

Binary Indexed Tree for prefix sum queries.

class FenwickTree {
    constructor(n) {
        this.tree = Array(n + 1).fill(0);
    }
    update(i, delta) {
        while (i < this.tree.length) {
            this.tree[i] += delta;
            i += i & (-i);
        }
    }
    query(i) {
        let sum = 0;
        while (i > 0) {
            sum += this.tree[i];
            i -= i & (-i);
        }
        return sum;
    }
}

Trie (Prefix Tree)

O(m)

Tree structure for efficient string operations.

class TrieNode {
    constructor() {
        this.children = {};
        this.isEnd = false;
    }
}
class Trie {
    insert(word) {
        let node = this.root;
        for (const char of word) {
            if (!node.children[char]) {
                node.children[char] = new TrieNode();
            }
            node = node.children[char];
        }
        node.isEnd = true;
    }
}

Suffix Array

O(n log n)

Sorted array of all suffixes for string processing.

function buildSuffixArray(s) {
    const suffixes = [];
    for (let i = 0; i < s.length; i++) {
        suffixes.push({ suffix: s.slice(i), index: i });
    }
    suffixes.sort((a, b) => 
        a.suffix.localeCompare(b.suffix)
    );
    return suffixes.map(x => x.index);
}
