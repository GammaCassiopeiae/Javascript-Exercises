Graph Algorithms

Visualize graph traversal and shortest path algorithms.

BFS

O(V+E)

Breadth-First Search explores level by level using a queue.

function BFS(graph, start) {
    const queue = [start];
    const visited = new Set([start]);
    while (queue.length) {
        const node = queue.shift();
        visit(node);
        graph[node].forEach(neighbor => {
            if (!visited.has(neighbor)) {
                visited.add(neighbor);
                queue.push(neighbor);
            }
        });
    }
}

DFS

O(V+E)

Depth-First Search explores as far as possible using recursion/stack.

function DFS(graph, node, visited) {
    if (visited.has(node)) return;
    visited.add(node);
    visit(node);
    graph[node].forEach(neighbor => {
        DFS(graph, neighbor, visited);
    });
}

Dijkstra

O(E log V)

Finds shortest path from source to all nodes using priority queue.

function dijkstra(graph, start) {
    const dist = {};
    const pq = new PriorityQueue();
    dist[start] = 0;
    pq.enqueue(start, 0);
    while (!pq.isEmpty()) {
        const node = pq.dequeue();
        graph[node].forEach(([neighbor, weight]) => {
            if (dist[node] + weight < (dist[neighbor] || Infinity)) {
                dist[neighbor] = dist[node] + weight;
                pq.enqueue(neighbor, dist[neighbor]);
            }
        });
    }
}

A* Search

O(E log V)

Finds shortest path using heuristic function f(n) = g(n) + h(n).

function AStar(graph, start, goal) {
    const openSet = [start];
    const gScore = { [start]: 0 };
    const fScore = { [start]: heuristic(start, goal) };
    while (openSet.length) {
        const current = getNodeWithLowestFScore(openSet, fScore);
        if (current === goal) return reconstructPath();
        openSet.splice(openSet.indexOf(current), 1);
        graph[current].forEach(neighbor => {
            const tentativeG = gScore[current] + dist(current, neighbor);
            if (tentativeG < (gScore[neighbor] || Infinity)) {
                gScore[neighbor] = tentativeG;
                fScore[neighbor] = tentativeG + heuristic(neighbor, goal);
                openSet.push(neighbor);
            }
        });
    }
}
