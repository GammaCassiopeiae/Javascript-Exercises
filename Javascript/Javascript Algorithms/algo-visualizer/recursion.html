Factorial

O(n)

Classic recursion: n! = n × (n-1)! with base case 0! = 1.

function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

Tower of Hanoi

O(2ⁿ)

Move n disks from source to destination using auxiliary peg.

function hanoi(n, source, aux, dest) {
    if (n === 1) {
        move(source, dest);
        return;
    }
    hanoi(n-1, source, dest, aux);
    move(source, dest);
    hanoi(n-1, aux, source, dest);
}

Permutations

O(n!)

Generate all possible arrangements of n elements.

function permute(arr, start = 0) {
    if (start === arr.length - 1) {
        result.push([...arr]);
        return;
    }
    for (let i = start; i < arr.length; i++) {
        [arr[start], arr[i]] = [arr[i], arr[start]];
        permute(arr, start + 1);
        [arr[start], arr[i]] = [arr[i], arr[start]];
    }
}

Subsets (Power Set)

O(2ⁿ)

Generate all possible subsets of a set.

function subsets(nums, index = 0, current = []) {
    if (index === nums.length) {
        result.push([...current]);
        return;
    }
    // Include nums[index]
    current.push(nums[index]);
    subsets(nums, index + 1, current);
    // Exclude nums[index]
    current.pop();
    subsets(nums, index + 1, current);
}
