Searching Algorithms

Visualize how algorithms find elements in data structures.

Linear Search

O(n)

Sequentially checks each element until target is found.

function linearSearch(arr, target) {
    for (let i = 0; i < arr.length; i++) {
        if (arr[i] === target) return i;
    }
    return -1;
}

Binary Search

O(log n)

Efficiently finds target in sorted array by halving search space.

function binarySearch(arr, target) {
    let left = 0, right = arr.length - 1;
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        if (arr[mid] === target) return mid;
        if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    return -1;
}

Jump Search

O(√n)

Jumps ahead by fixed steps, then linear search in block.

function jumpSearch(arr, target) {
    const step = Math.floor(Math.sqrt(arr.length));
    let prev = 0;
    while (arr[Math.min(step, arr.length) - 1] < target) {
        prev = step;
        step += Math.floor(Math.sqrt(arr.length));
    }
    while (arr[prev] < target) prev++;
    return arr[prev] === target ? prev : -1;
}

Exponential Search

O(log n)

Finds range by doubling, then binary search within range.

function exponentialSearch(arr, target) {
    if (arr[0] === target) return 0;
    let i = 1;
    while (i < arr.length && arr[i] <= target) i *= 2;
    return binarySearch(arr.slice(i/2, Math.min(i+1, arr.length)), target);
}
