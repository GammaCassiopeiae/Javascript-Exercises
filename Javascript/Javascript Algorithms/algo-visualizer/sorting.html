Bubble Sort

O(n²)

Simple comparison-based algorithm that repeatedly swaps adjacent elements if they are in wrong order.

Unsorted
Comparing
Swap
Sorted 
function bubbleSort(arr) {
    for (let i = 0; i < arr.length; i++) {
        for (let j = 0; j < arr.length - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap elements
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
}

Quick Sort

O(n log n)

Divide-and-conquer algorithm that selects a pivot and partitions array around it.

function quickSort(arr, low, high) {
    if (low < high) {
        const pi = partition(arr, low, high);
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

Merge Sort

O(n log n)

Divide-and-conquer algorithm that divides array into halves, sorts them, and merges.

Unsorted
Merging
Sorted 
function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    
    return merge(left, right);
}

Heap Sort

O(n log n)

Comparison-based algorithm using binary heap data structure for efficient sorting.

function heapSort(arr) {
    const n = arr.length;
    
    // Build max heap
    for (let i = Math.floor(n / 2) - 1; i >= 0; i--)
        heapify(arr, n, i);
    
    // Extract elements
    for (let i = n - 1; i > 0; i--) {
        [arr[0], arr[i]] = [arr[i], arr[0]];
        heapify(arr, i, 0);
    }
}
