String Algorithms

Pattern matching and string processing algorithms.

KMP Algorithm

O(n+m)

Knuth-Morris-Pratt uses failure function for efficient pattern matching.

function KMP(text, pattern) {
    const lps = computeLPS(pattern);
    let i = 0, j = 0;
    while (i < text.length) {
        if (pattern[j] === text[i]) {
            i++; j++;
        }
        if (j === pattern.length) {
            return i - j; // Found
        } else if (i < text.length && pattern[j] !== text[i]) {
            if (j !== 0) j = lps[j - 1];
            else i++;
        }
    }
    return -1;
}

Rabin-Karp

O(n+m)

Uses rolling hash for efficient pattern matching.

function rabinKarp(text, pattern) {
    const d = 256, q = 101;
    const m = pattern.length, n = text.length;
    let p = 0, t = 0, h = 1;
    for (let i = 0; i < m - 1; i++) h = (h * d) % q;
    for (let i = 0; i < m; i++) {
        p = (d * p + pattern[i]) % q;
        t = (d * t + text[i]) % q;
    }
    for (let i = 0; i <= n - m; i++) {
        if (p === t && text.slice(i, i+m) === pattern) return i;
        if (i < n - m) {
            t = (d * (t - text[i] * h) + text[i + m]) % q;
            if (t < 0) t += q;
        }
    }
    return -1;
}

Z-Algorithm

O(n+m)

Computes Z-array for linear time pattern matching.

function zAlgorithm(text, pattern) {
    const concat = pattern + '$' + text;
    const z = computeZ(concat);
    for (let i = pattern.length + 1; i < concat.length; i++) {
        if (z[i] === pattern.length) {
            return i - pattern.length - 1;
        }
    }
    return -1;
}

Manacher's (Longest Palindrome)

O(n)

Finds longest palindromic substring in linear time.

function manacher(s) {
    const t = '#' + s.split('').join('#') + '#';
    const p = Array(t.length).fill(0);
    let c = 0, r = 0;
    for (let i = 0; i < t.length; i++) {
        const mirror = 2 * c - i;
        if (i < r) p[i] = Math.min(r - i, p[mirror]);
        while (t[i + p[i] + 1] === t[i - p[i] - 1]) p[i]++;
        if (i + p[i] > r) { c = i; r = i + p[i]; }
    }
    return Math.max(...p);
}
