Dynamic Programming Algorithms

1. Longest Increasing Subsequence

function lengthOfLIS(nums) { if (!nums.length) return 0; let dp = new Array(nums.length).fill(1); for (let i = 1; i < nums.length; i++) { for (let j = 0; j < i; j++) { if (nums[i] > nums[j]) dp[i] = Math.max(dp[i], dp[j] + 1); } } return Math.max(...dp); }

2. Maximum Subarray (Kadane's)

function maxSubArray(nums) { let currSum = nums[0], maxSum = nums[0]; for (let i = 1; i < nums.length; i++) { currSum = Math.max(nums[i], currSum + nums[i]); maxSum = Math.max(maxSum, currSum); } return maxSum; }

3. Coin Change

function coinChange(coins, amount) { let dp = new Array(amount + 1).fill(Infinity); dp[0] = 0; for (let coin of coins) { for (let i = coin; i <= amount; i++) { dp[i] = Math.min(dp[i], dp[i - coin] + 1); } } return dp[amount] === Infinity ? -1 : dp[amount]; }
