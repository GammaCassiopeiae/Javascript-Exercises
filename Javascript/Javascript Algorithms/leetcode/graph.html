1. Number of Islands (BFS/DFS)

function numIslands(grid) { if (!grid) return 0; let count = 0; for (let r = 0; r < grid.length; r++) { for (let c = 0; c < grid[0].length; c++) { if (grid[r][c] === '1') { count++; dfs(grid, r, c); } } } return count; } function dfs(grid, r, c) { if (r < 0 || c < 0 || r >= grid.length || c >= grid[0].length || grid[r][c] === '0') return; grid[r][c] = '0'; dfs(grid, r + 1, c); dfs(grid, r - 1, c); dfs(grid, r, c + 1); dfs(grid, r, c - 1); }

2. Max Area of Island

function maxAreaOfIsland(grid) { let maxArea = 0; for (let r = 0; r < grid.length; r++) { for (let c = 0; c < grid[0].length; c++) { if (grid[r][c] === 1) maxArea = Math.max(maxArea, areaDfs(grid, r, c)); } } return maxArea; } function areaDfs(grid, r, c) { if (r < 0 || c < 0 || r >= grid.length || c >= grid[0].length || grid[r][c] === 0) return 0; grid[r][c] = 0; return 1 + areaDfs(grid, r + 1, c) + areaDfs(grid, r - 1, c) + areaDfs(grid, r, c + 1) + areaDfs(grid, r, c - 1); }
