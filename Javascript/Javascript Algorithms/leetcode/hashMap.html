1. Two Sum

function twoSum(nums, target) { let map = new Map(); for (let i = 0; i < nums.length; i++) { let complement = target - nums[i]; if (map.has(complement)) return [map.get(complement), i]; map.set(nums[i], i); } }

2. Group Anagrams

function groupAnagrams(strs) { let map = {}; for (let s of strs) { let count = new Array(26).fill(0); for (let char of s) count[char.charCodeAt(0) - 97]++; let key = count.join('#'); if (!map[key]) map[key] = []; map[key].push(s); } return Object.values(map); }

3. Subarray Sum Equals K

function subarraySum(nums, k) { let count = 0, sum = 0; let map = new Map(); map.set(0, 1); for (let i = 0; i < nums.length; i++) { sum += nums[i]; if (map.has(sum - k)) count += map.get(sum - k); map.set(sum, (map.get(sum) || 0) + 1); } return count; }
