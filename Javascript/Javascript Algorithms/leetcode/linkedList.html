1. Linked List Cycle

function hasCycle(head) { let slow = head, fast = head; while (fast && fast.next) { slow = slow.next; fast = fast.next.next; if (slow === fast) return true; } return false; }

function detectCycle(head) { let slow = head, fast = head; while (fast && fast.next) { slow = slow.next; fast = fast.next.next; if (slow === fast) { let pointer = head; while (pointer !== slow) { pointer = pointer.next; slow = slow.next; } return pointer; } } return null; }

3. Remove Duplicates from Sorted List

function deleteDuplicates(head) { let curr = head; while (curr && curr.next) { if (curr.val === curr.next.val) { curr.next = curr.next.next; } else { curr = curr.next; } } return head; }
