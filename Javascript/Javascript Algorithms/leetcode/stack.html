1. Valid Parentheses

function isValid(s) { let stack = []; let mapping = { ')': '(', '}': '{', ']': '[' }; for (let char of s) { if (mapping[char]) { let topElement = stack.length === 0 ? '#' : stack.pop(); if (topElement !== mapping[char]) return false; } else { stack.push(char); } } return stack.length === 0; }

2. Reverse Linked List

function reverseList(head) { let prev = null, curr = head; while (curr) { let nextTemp = curr.next; curr.next = prev; prev = curr; curr = nextTemp; } return prev; }
