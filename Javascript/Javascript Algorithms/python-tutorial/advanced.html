[01] EXCEPTION HANDLING

ADVANCED MODULE_05

Try-Except Blocks

Exception handling allows you to handle runtime errors gracefully. Python uses try-except blocks to catch and handle exceptions.

exceptions.py PYTHON 
# Basic try-except
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")

# Multiple exceptions
try:
    num = int(input("Enter a number: "))
    result = 10 / num
except ValueError:
    print("Invalid input! Please enter a number.")
except ZeroDivisionError:
    print("Cannot divide by zero!")

# Catching multiple exceptions in one clause
try:
    risky_operation()
except (ValueError, TypeError, KeyError) as e:
    print(f"Error occurred: {e}")

# Try-except-else
try:
    num = int(input("Enter a number: "))
except ValueError:
    print("Invalid number!")
else:
    print(f"You entered: {num}")

# Try-except-finally
try:
    file = open("data.txt", "r")
    content = file.read()
except FileNotFoundError:
    print("File not found!")
finally:
    print("Execution complete")
    if "file" in locals():
        file.close()

Raising Exceptions

# Raising exceptions
def validate_age(age):
    if age < 0:
        raise ValueError("Age cannot be negative")
    if age > 150:
        raise ValueError("Age seems unrealistic")
    return f"Age validated: {age}"

try:
    print(validate_age(-5))
except ValueError as e:
    print(f"Validation error: {e}")

# Re-raising exceptions
try:
    1 / 0
except ZeroDivisionError as e:
    print("Caught exception")
    raise  # Re-raise the same exception

[03] USER-DEFINED EXCEPTIONS

Custom Exception Classes

You can create your own exception classes by inheriting from the built-in Exception class. 

# Custom exception base class
class MyCustomError(Exception):
    """Base class for custom exceptions"""
    pass

# Specific custom exception
class InvalidEmailError(MyCustomError):
    """Raised when email format is invalid"""
    def __init__(self, email, message="Invalid email format"):
        self.email = email
        self.message = message
        super().__init__(f"{self.message}: {self.email}")

class InsufficientFundsError(MyCustomError):
    """Raised when account has insufficient funds"""
    def __init__(self, balance, amount):
        self.balance = balance
        self.amount = amount
        super().__init__(
            f"Balance ${balance} insufficient for ${amount} transaction"
        )

def validate_email(email):
    if "@" not in email:
        raise InvalidEmailError(email)
    return "Email valid"

try:
    validate_email("invalid-email")
except InvalidEmailError as e:
    print(e)

[04] FILE HANDLING

Opening Files

Python provides built-in functions for file operations. Always close files after use or use context managers. 

# File modes
# 'r' - Read (default)
# 'w' - Write (overwrites)
# 'a' - Append
# 'x' - Exclusive creation
# 'b' - Binary mode
# 't' - Text mode (default)
# '+' - Read and write

# Basic file opening
file = open("example.txt", "r")
content = file.read()
file.close()

# Using context manager (recommended)
with open("example.txt", "r") as file:
    content = file.read()
# File automatically closed

# Reading with encoding
with open("unicode.txt", "r", encoding="utf-8") as file:
    content = file.read()

[05] READING FILES

Read Methods

# Read entire file
with open("file.txt", "r") as f:
    content = f.read()
    print(content)

# Read specific number of characters
with open("file.txt", "r") as f:
    content = f.read(100)  # Read first 100 chars

# Read line by line
with open("file.txt", "r") as f:
    for line in f:
        print(line.strip())

# Read all lines into list
with open("file.txt", "r") as f:
    lines = f.readlines()
    print(lines)

# Read single line
with open("file.txt", "r") as f:
    first_line = f.readline()
    second_line = f.readline()

# Read binary file
with open("image.png", "rb") as f:
    binary_data = f.read()

[06] WRITING FILES

Write and Append

# Write to file (overwrites existing)
with open("output.txt", "w") as f:
    f.write("Hello, World!\n")
    f.write("Second line\n")

# Write multiple lines
lines = ["Line 1\n", "Line 2\n", "Line 3\n"]
with open("output.txt", "w") as f:
    f.writelines(lines)

# Append to file
with open("output.txt", "a") as f:
    f.write("Appended line\n")

# Write with formatting
data = {"name": "Alice", "age": 30}
with open("data.txt", "w") as f:
    for key, value in data.items():
        f.write(f"{key}: {value}\n")

# Write binary data
data = bytes([0x00, 0x01, 0x02, 0x03])
with open("binary.bin", "wb") as f:
    f.write(data)

[07] OS MODULE

Operating System Interface

The os module provides functions for interacting with the operating system. 

import os

# Get current working directory
cwd = os.getcwd()
print(f"Current directory: {cwd}")

# Change directory
os.chdir("/path/to/directory")

# List directory contents
files = os.listdir(".")
print(files)

# Create directory
os.mkdir("new_folder")
os.makedirs("parent/child/grandchild")  # Recursive

# Remove directory
os.rmdir("empty_folder")
os.removedirs("parent/child/grandchild")

# Remove file
os.remove("file.txt")
os.unlink("file.txt")  # Same as remove

# Rename file/directory
os.rename("old_name.txt", "new_name.txt")

# Get file info
stat_info = os.stat("file.txt")
print(f"Size: {stat_info.st_size} bytes")

# Check if path exists
print(os.path.exists("file.txt"))
print(os.path.isfile("file.txt"))
print(os.path.isdir("folder"))

# Join paths
full_path = os.path.join("folder", "subfolder", "file.txt")

# Get absolute path
abs_path = os.path.abspath("relative/path")

# Split path
directory, filename = os.path.split("/path/to/file.txt")
name, ext = os.path.splitext("file.txt")

# Environment variables
path = os.environ.get("PATH")
os.environ["MY_VAR"] = "value"

# Run system command
os.system("dir")  # Windows
result = os.popen("dir").read()

[08] PATHLIB MODULE

Object-Oriented Path Handling

The pathlib module provides a more modern, object-oriented approach to handling filesystem paths. 

from pathlib import Path, PurePath

# Create Path object
p = Path("folder/file.txt")

# Current directory
cwd = Path.cwd()

# Home directory
home = Path.home()

# Path parts
print(p.parent)      # folder
print(p.name)        # file.txt
print(p.stem)        # file
print(p.suffix)      # .txt
print(p.anchor)      # Drive letter on Windows

# Join paths
new_path = p.parent / "new_folder" / "new_file.txt"

# Check existence
print(p.exists())
print(p.is_file())
print(p.is_dir())

# Read file
content = p.read_text()
binary = p.read_bytes()

# Write file
p.write_text("Hello, World!")
p.write_bytes(bytes([0, 1, 2]))

# Create directory
Path("new_folder").mkdir()
Path("a/b/c").mkdir(parents=True, exist_ok=True)

# Iterate directory
for item in Path(".").iterdir():
    print(item)

# Glob patterns
python_files = list(Path(".").glob("*.py"))
all_files = list(Path(".").rglob("*"))  # Recursive

# Move/rename
p.rename("new_name.txt")

# Delete
p.unlink()  # Delete file
p.parent.rmdir()  # Delete empty directory

[09] DATABASE HANDLING

SQLite Database

Python has built-in support for SQLite databases through the sqlite3 module. 

import sqlite3

# Connect to database (creates if doesn't exist)
conn = sqlite3.connect("example.db")
cursor = conn.cursor()

# Create table
cursor.execute("""
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        age INTEGER
    )
""")

# Insert data
cursor.execute(
    "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
    ("Alice", "alice@email.com", 30)
)

# Insert multiple
users = [
    ("Bob", "bob@email.com", 25),
    ("Charlie", "charlie@email.com", 35)
]
cursor.executemany(
    "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
    users
)

# Commit changes
conn.commit()

# Query data
cursor.execute("SELECT * FROM users")
all_users = cursor.fetchall()

cursor.execute("SELECT * FROM users WHERE age > ?", (28,))
filtered = cursor.fetchall()

# Update data
cursor.execute(
    "UPDATE users SET age = ? WHERE name = ?",
    (31, "Alice")
)

# Delete data
cursor.execute("DELETE FROM users WHERE age < ?", (30,))

conn.commit()

# Close connection
conn.close()

# Using context manager
with sqlite3.connect("example.db") as conn:
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM users")
    for row in cursor:
        print(row)

]
collection.insert_many(users)

for doc in collection.find({"age": {"$gt": 28}}):

collection.update_one(
    {"name": "Alice"},
    {"$set": {"age": 31}}
)

# Delete document
collection.delete_one({"name": "Bob"})
