  return {x: v.x / mag, y: v.y / mag};
}

08_CIRCLE_COLLISION

function checkCircle(c1, c2) {
  let dx = c1.x - c2.x;
  let dy = c1.y - c2.y;
  let dist = Math.sqrt(dx*dx + dy*dy);
  return dist < c1.r + c2.r;
}

L2-4: Izračuna dejansko razdaljo med središči dveh krogov.
