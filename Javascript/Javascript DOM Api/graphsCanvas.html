1. Line Chart

Line charts are used to display data points connected by straight line segments. They are ideal for showing trends over time.

function drawLineChart(ctx, data, labels, title) {
    // Clear canvas
    ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    
    // Set up title
    ctx.font = 'bold 16px Arial';
    ctx.fillStyle = '#333';
    ctx.fillText(title, 20, 30);
    
    // Set up axes
    const margin = 60;
    const width = ctx.canvas.width - margin * 2;
    const height = ctx.canvas.height - margin * 2;
    
    // Calculate scale
    const xScale = width / (data.length - 1);
    const yScale = height / Math.max(...data);
    
    // Draw axes
    ctx.beginPath();
    ctx.moveTo(margin, margin);
    ctx.lineTo(margin, margin + height);
    ctx.lineTo(margin + width, margin + height);
    ctx.strokeStyle = '#ccc';
    ctx.stroke();
    
    // Draw data points and lines
    ctx.beginPath();
    ctx.moveTo(margin, margin + height - data[0] * yScale);
    
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * xScale;
        const y = margin + height - data[i] * yScale;
        
        // Draw point
        ctx.moveTo(x, y);
        ctx.arc(x, y, 3, 0, 2 * Math.PI);
        
        // Draw label
        if (i === 0 || i === data.length - 1) {
            ctx.font = '12px Arial';
            ctx.fillStyle = '#333';
            ctx.fillText(labels[i], x, margin + height + 20);
        }
    }
    
    ctx.strokeStyle = '#667eea';
    ctx.lineWidth = 2;
    ctx.stroke();
    
    // Draw labels
    ctx.font = '12px Arial';
    ctx.fillStyle = '#666';
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * xScale;
        const y = margin + height - data[i] * yScale;
        
        // Draw point
        ctx.beginPath();
        ctx.arc(x, y, 4, 0, 2 * Math.PI);
        ctx.fillStyle = '#667eea';
        ctx.fill();
    }
}

2. Bar Chart

Bar charts display data using rectangular bars of different heights. They're great for comparing values across categories.

function drawBarChart(ctx, data, labels, title) {
    // Clear canvas
    ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    
    // Set up title
    ctx.font = 'bold 16px Arial';
    ctx.fillStyle = '#333';
    ctx.fillText(title, 20, 30);
    
    // Set up axes
    const margin = 60;
    const width = ctx.canvas.width - margin * 2;
    const height = ctx.canvas.height - margin * 2;
    
    // Calculate scale
    const barWidth = width / data.length;
    const maxData = Math.max(...data);
    const yScale = height / maxData;
    
    // Draw bars
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * barWidth;
        const y = margin + height - data[i] * yScale;
        const barHeight = data[i] * yScale;
        
        // Draw bar
        ctx.fillStyle = '#4CAF50';
        ctx.fillRect(x, y, barWidth - 5, barHeight);
        
        // Draw label
        ctx.font = '12px Arial';
        ctx.fillStyle = '#333';
        ctx.fillText(labels[i], x + 5, y - 5);
    }
    
    // Draw x-axis labels
    ctx.font = '12px Arial';
    ctx.fillStyle = '#666';
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * barWidth;
        ctx.fillText(labels[i], x, margin + height + 20);
    }
}

3. Pie Chart

Pie charts show proportions of a whole. Each slice represents a category's contribution to the total.

function drawPieChart(ctx, data, labels, title) {
    // Clear canvas
    ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    
    // Set up title
    ctx.font = 'bold 16px Arial';
    ctx.fillStyle = '#333';
    ctx.fillText(title, 20, 30);
    
    // Calculate total
    const total = data.reduce((sum, val) => sum + val, 0);
    
    // Set up center and radius
    const centerX = ctx.canvas.width / 2;
    const centerY = ctx.canvas.height / 2;
    const radius = Math.min(centerX, centerY) - 40;
    
    // Calculate start angle
    let startAngle = -Math.PI / 2;
    
    // Draw slices
    for (let i = 0; i < data.length; i++) {
        const sliceAngle = (data[i] / total) * 2 * Math.PI;
        const endAngle = startAngle + sliceAngle;
        
        // Set color
        const colors = ['#FF5733', '#33FF57', '#3357FF', '#F3FF33', '#FF33F3', '#33FFC7'];
        const color = colors[i % colors.length];
        
        // Draw slice
        ctx.beginPath();
        ctx.moveTo(centerX, centerY);
        ctx.arc(centerX, centerY, radius, startAngle, endAngle);
        ctx.closePath();
        
        // Fill slice
        ctx.fillStyle = color;
        ctx.fill();
        
        // Draw label
        const labelAngle = startAngle + sliceAngle / 2;
        const labelX = centerX + Math.cos(labelAngle) * radius * 0.7;
        const labelY = centerY + Math.sin(labelAngle) * radius * 0.7;
        
        ctx.font = '14px Arial';
        ctx.fillStyle = '#333';
        ctx.fillText(labels[i], labelX, labelY);
        
        startAngle = endAngle;
    }
}

4. Scatter Plot

Scatter plots show the relationship between two variables. Each point represents a pair of values.

function drawScatterPlot(ctx, xData, yData, title) {
    // Clear canvas
    ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    
    // Set up title
    ctx.font = 'bold 16px Arial';
    ctx.fillStyle = '#333';
    ctx.fillText(title, 20, 30);
    
    // Set up axes
    const margin = 60;
    const width = ctx.canvas.width - margin * 2;
    const height = ctx.canvas.height - margin * 2;
    
    // Calculate scales
    const xMin = Math.min(...xData);
    const xMax = Math.max(...xData);
    const yMin = Math.min(...yData);
    const yMax = Math.max(...yData);
    
    const xScale = width / (xMax - xMin);
    const yScale = height / (yMax - yMin);
    
    // Draw axes
    ctx.beginPath();
    ctx.moveTo(margin, margin + height);
    ctx.lineTo(margin, margin);
    ctx.lineTo(margin + width, margin);
    ctx.strokeStyle = '#ccc';
    ctx.stroke();
    
    // Draw data points
    ctx.beginPath();
    for (let i = 0; i < xData.length; i++) {
        const x = margin + (xData[i] - xMin) * xScale;
        const y = margin + height - (yData[i] - yMin) * yScale;
        
        // Draw point
        ctx.moveTo(x, y);
        ctx.arc(x, y, 4, 0, 2 * Math.PI);
        ctx.fillStyle = '#FF5733';
        ctx.fill();
    }
    
    // Draw labels
    ctx.font = '12px Arial';
    ctx.fillStyle = '#666';
    for (let i = 0; i < xData.length; i++) {
        const x = margin + (xData[i] - xMin) * xScale;
        const y = margin + height - (yData[i] - yMin) * yScale;
        
        // Draw point
        ctx.beginPath();
        ctx.arc(x, y, 4, 0, 2 * Math.PI);
        ctx.fillStyle = '#FF5733';
        ctx.fill();
    }
}

5. Area Chart

Area charts are similar to line charts but the area under the line is filled with color.

function drawAreaChart(ctx, data, labels, title) {
    // Clear canvas
    ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
    
    // Set up title
    ctx.font = 'bold 16px Arial';
    ctx.fillStyle = '#333';
    ctx.fillText(title, 20, 30);
    
    // Set up axes
    const margin = 60;
    const width = ctx.canvas.width - margin * 2;
    const height = ctx.canvas.height - margin * 2;
    
    // Calculate scale
    const xScale = width / (data.length - 1);
    const yScale = height / Math.max(...data);
    
    // Draw area
    ctx.beginPath();
    ctx.moveTo(margin, margin + height);
    
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * xScale;
        const y = margin + height - data[i] * yScale;
        
        ctx.lineTo(x, y);
    }
    
    ctx.lineTo(margin + width, margin + height);
    ctx.closePath();
    
    // Fill area
    ctx.fillStyle = 'rgba(102, 126, 234, 0.3)';
    ctx.fill();
    
    // Draw line
    ctx.beginPath();
    ctx.moveTo(margin, margin + height - data[0] * yScale);
    
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * xScale;
        const y = margin + height - data[i] * yScale;
        
        ctx.lineTo(x, y);
    }
    
    ctx.strokeStyle = '#667eea';
    ctx.lineWidth = 2;
    ctx.stroke();
    
    // Draw labels
    ctx.font = '12px Arial';
    ctx.fillStyle = '#333';
    for (let i = 0; i < data.length; i++) {
        const x = margin + i * xScale;
        const y = margin + height - data[i] * yScale;
        
        // Draw point
        ctx.beginPath();
        ctx.arc(x, y, 4, 0, 2 * Math.PI);
        ctx.fillStyle = '#667eea';
        ctx.fill();
    }
}
