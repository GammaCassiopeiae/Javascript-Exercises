#62 UNIQUE PATHS

Find the number of ways to reach the bottom-right corner of an m x n grid.

def uniquePaths(m: int, n: int) -> int:
    row = [1] * n
    for i in range(m - 1):
        newRow = [1] * n
        for j in range(n - 2, -1, -1):
            newRow[j] = newRow[j + 1] + row[j]
        row = newRow
    return row[0]

Complexity: O(m * n) Time | O(n) Space

#1143 LONGEST COMMON SUBSEQUENCE

Find the length of the longest subsequence shared by two strings.

def longestCommonSubsequence(text1: str, text2: str) -> int:
    dp = [[0 for j in range(len(text2) + 1)] for i in range(len(text1) + 1)]
    for i in range(len(text1) - 1, -1, -1):
        for j in range(len(text2) - 1, -1, -1):
            if text1[i] == text2[j]:
                dp[i][j] = 1 + dp[i + 1][j + 1]
            else:
                dp[i][j] = max(dp[i][j + 1], dp[i + 1][j])
    return dp[0][0]

Complexity: O(n * m) Time | O(n * m) Space

#309 STOCK WITH COOLDOWN

Maximize profit with a 1-day cooldown after selling.

def maxProfit(prices: list[int]) -> int:
    dp = {} # (index, buying) -> max_profit
    def dfs(i, buying):
        if i >= len(prices): return 0
        if (i, buying) in dp: return dp[(i, buying)]
        
        cooldown = dfs(i + 1, buying)
        if buying:
            buy = dfs(i + 1, not buying) - prices[i]
            dp[(i, buying)] = max(buy, cooldown)
        else:
            sell = dfs(i + 2, not buying) + prices[i]
            dp[(i, buying)] = max(sell, cooldown)
        return dp[(i, buying)]
    return dfs(0, True)

Complexity: O(n) Time | O(n) Space

#97 INTERLEAVING STRING

Determine if s3 is formed by interleaving s1 and s2.

def isInterleave(s1: str, s2: str, s3: str) -> bool:
    if len(s1) + len(s2) != len(s3): return False
    dp = [[False] * (len(s2) + 1) for i in range(len(s1) + 1)]
    dp[len(s1)][len(s2)] = True
    for i in range(len(s1), -1, -1):
        for j in range(len(s2), -1, -1):
            if i < len(s1) and s1[i] == s3[i + j] and dp[i + 1][j]:
                dp[i][j] = True
            if j < len(s2) and s2[j] == s3[i + j] and dp[i][j + 1]:
                dp[i][j] = True
    return dp[0][0]

Complexity: O(n * m) Time | O(n * m) Space

#72 EDIT DISTANCE

Find min operations (insert, delete, replace) to convert word1 to word2.

def minDistance(word1: str, word2: str) -> int:
    dp = [[0] * (len(word2) + 1) for i in range(len(word1) + 1)]
    for j in range(len(word2) + 1): dp[len(word1)][j] = len(word2) - j
    for i in range(len(word1) + 1): dp[i][len(word2)] = len(word1) - i
    for i in range(len(word1) - 1, -1, -1):
        for j in range(len(word2) - 1, -1, -1):
            if word1[i] == word2[j]:
                dp[i][j] = dp[i + 1][j + 1]
            else:
                dp[i][j] = 1 + min(dp[i + 1][j], dp[i][j + 1], dp[i + 1][j + 1])
    return dp[0][0]

Complexity: O(n * m) Time | O(n * m) Space

#494 TARGET SUM

Count ways to assign +/- to elements to reach a target sum.

def findTargetSumWays(nums: list[int], target: int) -> int:
    dp = {} # (index, total) -> ways
    def backtrack(i, total):
        if i == len(nums):
            return 1 if total == target else 0
        if (i, total) in dp:
            return dp[(i, total)]
        dp[(i, total)] = backtrack(i + 1, total + nums[i]) + \
                         backtrack(i + 1, total - nums[i])
        return dp[(i, total)]
    return backtrack(0, 0)

Complexity: O(n * sum) Time | O(n * sum) Space

#377 COMBINATION SUM IV

Find the number of possible combinations that add up to a target.

def combinationSum4(nums: list[int], target: int) -> int:
    dp = {0: 1}
    for i in range(1, target + 1):
        dp[i] = 0
        for n in nums:
            dp[i] += dp.get(i - n, 0)
    return dp[target]

Complexity: O(target * n) Time | O(target) Space

#329 LONGEST INCREASING PATH

Find the length of the longest increasing path in an m x n matrix.

def longestIncreasingPath(matrix: list[list[int]]) -> int:
    ROWS, COLS = len(matrix), len(matrix[0])
    dp = {} # (r, c) -> LIP
    def dfs(r, c, prevVal):
        if (r < 0 or r == ROWS or c < 0 or c == COLS or matrix[r][c] <= prevVal):
            return 0
        if (r, c) in dp: return dp[(r, c)]
        res = 1
        res = max(res, 1 + dfs(r + 1, c, matrix[r][c]))
        res = max(res, 1 + dfs(r - 1, c, matrix[r][c]))
        res = max(res, 1 + dfs(r, c + 1, matrix[r][c]))
        res = max(res, 1 + dfs(r, c - 1, matrix[r][c]))
        dp[(r, c)] = res
        return res
    for r in range(ROWS):
        for c in range(COLS):
            dfs(r, c, -1)
    return max(dp.values())

Complexity: O(n * m) Time | O(n * m) Space

#115 DISTINCT SUBSEQUENCES

Count number of distinct subsequences of s which equals t.

def numDistinct(s: str, t: str) -> int:
    dp = {}
    def dfs(i, j):
        if j == len(t): return 1
        if i == len(s): return 0
        if (i, j) in dp: return dp[(i, j)]
        if s[i] == t[j]:
            dp[(i, j)] = dfs(i + 1, j + 1) + dfs(i + 1, j)
        else:
            dp[(i, j)] = dfs(i + 1, j)
        return dp[(i, j)]
    return dfs(0, 0)

Complexity: O(n * m) Time | O(n * m) Space

#312 BURST BALLOONS

Maximize coins by bursting balloons strategically.

def maxCoins(nums: list[int]) -> int:
    nums = [1] + nums + [1]
    dp = {}
    def dfs(l, r):
        if l > r: return 0
        if (l, r) in dp: return dp[(l, r)]
        dp[(l, r)] = 0
        for i in range(l, r + 1):
            coins = nums[l-1] * nums[i] * nums[r+1]
            coins += dfs(l, i - 1) + dfs(i + 1, r)
            dp[(l, r)] = max(dp[(l, r)], coins)
        return dp[(l, r)]
    return dfs(1, len(nums) - 2)

Complexity: O(n³) Time | O(n²) Space

#10 REGULAR EXPRESSION MATCHING

Implement regex matching with '.' and '*'.

def isMatch(s: str, p: str) -> bool:
    cache = {}
    def dfs(i, j):
        if (i, j) in cache: return cache[(i, j)]
        if i >= len(s) and j >= len(p): return True
        if j >= len(p): return False
        
        match = i < len(s) and (s[i] == p[j] or p[j] == ".")
        if (j + 1) < len(p) and p[j + 1] == "*":
            cache[(i, j)] = dfs(i, j + 2) or (match and dfs(i + 1, j))
            return cache[(i, j)]
        if match:
            cache[(i, j)] = dfs(i + 1, j + 1)
            return cache[(i, j)]
        cache[(i, j)] = False
        return False
    return dfs(0, 0)

Complexity: O(n * m) Time | O(n * m) Space

#63 UNIQUE PATHS II

Unique paths in a grid with obstacles.

def uniquePathsWithObstacles(obstacleGrid: list[list[int]]) -> int:
    M, N = len(obstacleGrid), len(obstacleGrid[0])
    dp = [0] * N
    dp[N-1] = 1
    for r in range(M - 1, -1, -1):
        for c in range(N - 1, -1, -1):
            if obstacleGrid[r][c]:
                dp[c] = 0
            elif c + 1 < N:
                dp[c] = dp[c] + dp[c + 1]
    return dp[0]

Complexity: O(M * N) Time | O(N) Space

#64 MINIMUM PATH SUM

Find path from top-left to bottom-right that minimizes sum.

def minPathSum(grid: list[list[int]]) -> int:
    ROWS, COLS = len(grid), len(grid[0])
    res = [[float("inf")] * (COLS + 1) for r in range(ROWS + 1)]
    res[ROWS - 1][COLS] = 0
    for r in range(ROWS - 1, -1, -1):
        for c in range(COLS - 1, -1, -1):
            res[r][c] = grid[r][c] + min(res[r + 1][c], res[r][c + 1])
    return res[0][0]

Complexity: O(M * N) Time | O(M * N) Space

#518 COIN CHANGE II

Find the number of combinations that make up the amount.

def change(amount: int, coins: list[int]) -> int:
    dp = [0] * (amount + 1)
    dp[0] = 1
    for coin in coins:
        for i in range(coin, amount + 1):
            dp[i] += dp[i - coin]
    return dp[amount]

Complexity: O(n * amount) Time | O(amount) Space

#44 WILDCARD MATCHING

Implement wildcard matching with '?' and '*'.

def isMatch(s: str, p: str) -> bool:
    dp = [[False] * (len(p) + 1) for _ in range(len(s) + 1)]
    dp[0][0] = True
    for j in range(1, len(p) + 1):
        if p[j-1] == '*': dp[0][j] = dp[0][j-1]
    for i in range(1, len(s) + 1):
        for j in range(1, len(p) + 1):
            if p[j-1] == s[i-1] or p[j-1] == '?':
                dp[i][j] = dp[i-1][j-1]
            elif p[j-1] == '*':
                dp[i][j] = dp[i-1][j] or dp[i][j-1]
    return dp[len(s)][len(p)]

Complexity: O(n * m) Time | O(n * m) Space

#123 STOCK III (MAX 2 TRANS)

Maximize profit with at most two transactions.

def maxProfit(prices: list[int]) -> int:
    if not prices: return 0
    buy1, buy2 = float('inf'), float('inf')
    sell1, sell2 = 0, 0
    for p in prices:
        buy1 = min(buy1, p)
        sell1 = max(sell1, p - buy1)
        buy2 = min(buy2, p - sell1)
        sell2 = max(sell2, p - buy2)
    return sell2

Complexity: O(n) Time | O(1) Space

#188 STOCK IV (MAX K TRANS)

Maximize profit with at most k transactions.

def maxProfit(k: int, prices: list[int]) -> int:
    if not prices: return 0
    if k >= len(prices) // 2:
        return sum(max(0, prices[i] - prices[i-1]) for i in range(1, len(prices)))
    dp = [[0] * len(prices) for _ in range(k + 1)]
    for i in range(1, k + 1):
        tmpMax = -prices[0]
        for j in range(1, len(prices)):
            dp[i][j] = max(dp[i][j-1], prices[j] + tmpMax)
            tmpMax = max(tmpMax, dp[i-1][j-1] - prices[j])
    return dp[k][-1]

Complexity: O(k * n) Time | O(k * n) Space

#712 MIN ASCII DELETE SUM

Find min ASCII sum of deleted characters to make strings equal.

def minimumDeleteSum(s1: str, s2: str) -> int:
    m, n = len(s1), len(s2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(1, m + 1): dp[i][0] = dp[i-1][0] + ord(s1[i-1])
    for j in range(1, n + 1): dp[0][j] = dp[0][j-1] + ord(s2[j-1])
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = min(dp[i-1][j] + ord(s1[i-1]), dp[i][j-1] + ord(s2[j-1]))
    return dp[m][n]

Complexity: O(m * n) Time | O(m * n) Space

#174 DUNGEON GAME

Determine knight's minimum initial health to reach the princess.

def calculateMinimumHP(dungeon: list[list[int]]) -> int:
    ROWS, COLS = len(dungeon), len(dungeon[0])
    dp = [[float("inf")] * (COLS + 1) for _ in range(ROWS + 1)]
    dp[ROWS][COLS-1] = dp[ROWS-1][COLS] = 1
    for r in range(ROWS - 1, -1, -1):
        for c in range(COLS - 1, -1, -1):
            min_health = min(dp[r+1][c], dp[r][c+1]) - dungeon[r][c]
            dp[r][c] = max(1, min_health)
    return dp[0][0]

Complexity: O(M * N) Time | O(M * N) Space
