BIT_MANIPULATION

Perform low-level data processing using binary operations.

#136 SINGLE NUMBER

Find the element that appears only once in an array where every other element appears twice.

def singleNumber(nums: list[int]) -> int:
    res = 0
    for n in nums:
        res ^= n
    return res

Complexity: O(n) Time | O(1) Space

#191 NUMBER OF 1 BITS

Count the number of set bits (Hamming weight) in an integer.

def hammingWeight(n: int) -> int:
    res = 0
    while n:
        n &= (n - 1)
        res += 1
    return res

Complexity: O(1) Time | O(1) Space

#338 COUNTING BITS

Return an array where each element is the number of 1s in the binary representation of i.

def countBits(n: int) -> list[int]:
    dp = [0] * (n + 1)
    offset = 1
    for i in range(1, n + 1):
        if offset * 2 == i:
            offset = i
        dp[i] = 1 + dp[i - offset]
    return dp

Complexity: O(n) Time | O(1) Space (ignoring output)

#190 REVERSE BITS

Reverse bits of a given 32 bits unsigned integer.

def reverseBits(n: int) -> int:
    res = 0
    for i in range(32):
        bit = (n >> i) & 1
        res = res | (bit << (31 - i))
    return res

Complexity: O(1) Time | O(1) Space

#268 MISSING NUMBER

Find the only number in the range [0, n] that is missing from the array.

def missingNumber(nums: list[int]) -> int:
    res = len(nums)
    for i in range(len(nums)):
        res += (i - nums[i])
    return res

Complexity: O(n) Time | O(1) Space

#371 SUM OF TWO INTEGERS

Sum two integers without using + or - operators.

def getSum(a: int, b: int) -> int:
    mask = 0xffffffff
    while (b & mask) > 0:
        carry = (a & b) << 1
        a = (a ^ b)
        b = carry
    return (a & mask) if b > 0 else a

Complexity: O(1) Time | O(1) Space

#7 REVERSE INTEGER

Reverse digits of a signed 32-bit integer.

def reverse(x: int) -> int:
    MIN = -2147483648
    MAX = 2147483647
    res = 0
    while x:
        digit = int(math.fmod(x, 10))
        x = int(x / 10)
        if (res > MAX // 10 or (res == MAX // 10 and digit > MAX % 10)): return 0
        if (res < MIN // 10 or (res == MIN // 10 and digit < MIN % 10)): return 0
        res = (res * 10) + digit
    return res

Complexity: O(log n) Time | O(1) Space

#137 SINGLE NUMBER II

Find the element that appears once where others appear three times.

def singleNumber(nums: list[int]) -> int:
    ones, twos = 0, 0
    for n in nums:
        ones = (ones ^ n) & ~twos
        twos = (twos ^ n) & ~ones
    return ones

Complexity: O(n) Time | O(1) Space

#260 SINGLE NUMBER III

Find the two elements that appear only once.

def singleNumber(nums: list[int]) -> list[int]:
    xor = 0
    for n in nums: xor ^= n
    diff = xor & -xor
    res = [0, 0]
    for n in nums:
        if n & diff: res[0] ^= n
        else: res[1] ^= n
    return res

Complexity: O(n) Time | O(1) Space

#389 FIND THE DIFFERENCE

Find the character that was added to string t.

def findTheDifference(s: str, t: str) -> str:
    res = 0
    for c in s + t: res ^= ord(c)
    return chr(res)

Complexity: O(n) Time | O(1) Space

#461 HAMMING DISTANCE

Find the number of positions at which the corresponding bits are different.

def hammingDistance(x: int, y: int) -> int:
    xor = x ^ y
    res = 0
    while xor:
        res += xor & 1
        xor >>= 1
    return res

Complexity: O(1) Time | O(1) Space

#476 NUMBER COMPLEMENT

Output the complement number of a given integer.

def findComplement(num: int) -> int:
    mask = 1
    while mask < num:
        mask = (mask << 1) | 1
    return num ^ mask

Complexity: O(log n) Time | O(1) Space

#169 MAJORITY ELEMENT

Find the element that appears more than n/2 times.

def majorityElement(nums: list[int]) -> int:
    res, count = 0, 0
    for n in nums:
        if count == 0: res = n
        count += (1 if n == res else -1)
    return res

Complexity: O(n) Time | O(1) Space

#89 GRAY CODE

Return a n-bit gray code sequence.

def grayCode(n: int) -> list[int]:
    res = []
    for i in range(1 << n):
        res.append(i ^ (i >> 1))
    return res

Complexity: O(2^n) Time | O(2^n) Space

#201 BITWISE AND OF RANGE

Find bitwise AND of all numbers in range [left, right].

def rangeBitwiseAnd(left: int, right: int) -> int:
    shift = 0
    while left < right:
        left >>= 1
        right >>= 1
        shift += 1
    return left << shift

Complexity: O(log n) Time | O(1) Space
