#1 TWO SUM

Given an array of integers and a target, find indices of the two numbers that add up to the target.

def twoSum(nums: list[int], target: int) -> list[int]:
    prevMap = {}  # val : index

    for i, n in enumerate(nums):
        diff = target - n
        if diff in prevMap:
            return [prevMap[diff], i]
        prevMap[n] = i
    return []

Complexity: O(n) Time | O(n) Space

#217 CONTAINS DUPLICATE

Determine if any value appears at least twice in an array of integers.

def containsDuplicate(nums: list[int]) -> bool:
    hashset = set()

    for n in nums:
        if n in hashset:
            return True
        hashset.add(n)
    return False

Complexity: O(n) Time | O(n) Space

#242 VALID ANAGRAM

Check if two strings are anagrams of each other using frequency mapping.

def isAnagram(s: str, t: str) -> bool:
    if len(s) != len(t):
        return False

    countS, countT = {}, {}

    for i in range(len(s)):
        countS[s[i]] = 1 + countS.get(s[i], 0)
        countT[t[i]] = 1 + countT.get(t[i], 0)
    
    return countS == countT

Complexity: O(s + t) Time | O(s + t) Space

#1299 REPLACE ELEMENTS

Replace every element with the greatest element among the elements to its right, and replace the last element with -1.

def replaceElements(arr: list[int]) -> list[int]:
    rightMax = -1

    for i in range(len(arr) - 1, -1, -1):
        newMax = max(rightMax, arr[i])
        arr[i] = rightMax
        rightMax = newMax
    return arr

Complexity: O(n) Time | O(1) Space

#392 IS SUBSEQUENCE

Given two strings s and t, return true if s is a subsequence of t, or false otherwise.

def isSubsequence(s: str, t: str) -> bool:
    i, j = 0, 0
    while i < len(s) and j < len(t):
        if s[i] == t[j]:
            i += 1
        j += 1
    return i == len(s)

Complexity: O(t) Time | O(1) Space

def lengthOfLastWord(s: str) -> int:
    i, length = len(s) - 1, 0

    while s[i] == " ":
        i -= 1
    while i >= 0 and s[i] != " ":
        length += 1
        i -= 1
    return length

#118 PASCAL'S TRIANGLE

def generate(numRows: int) -> list[list[int]]:
    res = [[1]]

    for i in range(numRows - 1):
        temp = [0] + res[-1] + [0]
        row = []

            row.append(temp[j] + temp[j + 1])
        res.append(row)
    return res

#205 ISOMORPHIC STRINGS

Determine if two strings are isomorphic.

def isIsomorphic(s: str, t: str) -> bool:
    mapST, mapTS = {}, {}

    for c1, c2 in zip(s, t):
        if ((c1 in mapST and mapST[c1] != c2) or
            (c2 in mapTS and mapTS[c2] != c1)):
            return False
        mapST[c1] = c2
        mapTS[c2] = c1
    return True

Complexity: O(n) Time | O(n) Space

#169 MAJORITY ELEMENT

Find the element that appears more than n/2 times.

def majorityElement(nums: list[int]) -> int:
    res, count = 0, 0

    for n in nums:
        if count == 0:
            res = n
        count += (1 if n == res else -1)
    return res

Complexity: O(n) Time | O(1) Space

#496 NEXT GREATER ELEMENT I

Find the next greater element for each value in nums1 in the corresponding position of nums2.

def nextGreaterElement(nums1: list[int], nums2: list[int]) -> list[int]:
    nums1Idx = { n:i for i, n in enumerate(nums1) }
    res = [-1] * len(nums1)

    stack = []
    for i in range(len(nums2)):
        cur = nums2[i]
        while stack and cur > stack[-1]:
            val = stack.pop()
            idx = nums1Idx[val]
            res[idx] = cur
        if cur in nums1Idx:
            stack.append(cur)
    return res

Complexity: O(n + m) Time | O(m) Space

#724 FIND PIVOT INDEX

Find the index where the sum of numbers to the left equals the sum to the right.

def pivotIndex(nums: list[int]) -> int:
    total = sum(nums)
    leftSum = 0

    for i, x in enumerate(nums):
        if leftSum == (total - leftSum - x):
            return i
        leftSum += x
    return -1

Complexity: O(n) Time | O(1) Space

#303 RANGE SUM QUERY

Calculate the sum of elements between indices left and right inclusive.

class NumArray:
    def __init__(self, nums: list[int]):
        self.prefix = []
        cur = 0
        for n in nums:
            cur += n
            self.prefix.append(cur)

    def sumRange(self, left: int, right: int) -> int:
        r = self.prefix[right]
        l = self.prefix[left - 1] if left > 0 else 0
        return r - l

Complexity: O(1) Query Time | O(n) Init Time

#347 TOP K FREQUENT

Find the k most frequent elements in an array.

def topKFrequent(nums: list[int], k: int) -> list[int]:
    count = {}
    freq = [[] for i in range(len(nums) + 1)]

    for n in nums:
        count[n] = 1 + count.get(n, 0)
    for n, c in count.items():
        freq[c].append(n)

    res = []
    for i in range(len(freq) - 1, 0, -1):
        for n in freq[i]:
            res.append(n)
            if len(res) == k:
                return res

Complexity: O(n) Time | O(n) Space

#238 PRODUCT EXCEPT SELF

Return an array where each element is the product of all other elements without using division.

def productExceptSelf(nums: list[int]) -> list[int]:
    res = [1] * len(nums)

    prefix = 1
    for i in range(len(nums)):
        res[i] = prefix
        prefix *= nums[i]
    
    postfix = 1
    for i in range(len(nums) - 1, -1, -1):
        res[i] *= postfix
        postfix *= nums[i]
    return res

Complexity: O(n) Time | O(1) Space

#36 VALID SUDOKU

Determine if a 9 x 9 Sudoku board is valid.

import collections

def isValidSudoku(board: list[list[str]]) -> bool:
    cols = collections.defaultdict(set)
    rows = collections.defaultdict(set)
    squares = collections.defaultdict(set)

    for r in range(9):
        for c in range(9):
            if board[r][c] == ".":
                continue
            if (board[r][c] in rows[r] or
                board[r][c] in cols[c] or
                board[r][c] in squares[(r // 3, c // 3)]):
                return False
            cols[c].add(board[r][c])
            rows[r].add(board[r][c])
            squares[(r // 3, c // 3)].add(board[r][c])
    return True

Complexity: O(1) Time | O(1) Space (fixed board size)

#271 ENCODE & DECODE

Design an algorithm to encode a list of strings to a single string and decode it back.

def encode(strs):
    res = ""
    for s in strs:
        res += str(len(s)) + "#" + s
    return res

def decode(s):
    res, i = [], 0
    while i < len(s):
        j = i
        while s[j] != "#":
            j += 1
        length = int(s[i:j])
        res.append(s[j + 1 : j + 1 + length])
        i = j + 1 + length
    return res

Complexity: O(n) Time | O(1) Space

#128 LONGEST CONSECUTIVE

Find the length of the longest consecutive elements sequence.

def longestConsecutive(nums: list[int]) -> int:
    numSet = set(nums)
    longest = 0

    for n in nums:
        if (n - 1) not in numSet:
            length = 1
            while (n + length) in numSet:
                length += 1
            longest = max(length, longest)
    return longest

Complexity: O(n) Time | O(n) Space

#75 SORT COLORS

Sort an array with 0s, 1s, and 2s in-place (Dutch National Flag algorithm).

def sortColors(nums: list[int]) -> None:
    l, i, r = 0, 0, len(nums) - 1

    while i <= r:
        if nums[i] == 0:
            nums[l], nums[i] = nums[i], nums[l]
            l += 1
            i += 1
        elif nums[i] == 2:
            nums[i], nums[r] = nums[r], nums[i]
            r -= 1
        else:
            i += 1

Complexity: O(n) Time | O(1) Space

#692 TOP K FREQUENT WORDS

Find the k most frequent strings, sorted by frequency and then lexicographically.

import heapq
from collections import Counter

def topKFrequent(words: list[str], k: int) -> list[str]:
    counts = Counter(words)
    h = [(-count, word) for word, count in counts.items()]
    heapq.heapify(h)
    return [heapq.heappop(h)[1] for _ in range(k)]

Complexity: O(n + k log n) Time | O(n) Space

#2017 GRID GAME

Find the minimum score the second robot can get if the first robot plays optimally.

def gridGame(grid: list[list[int]]) -> int:
    n = len(grid[0])
    pre1, pre2 = grid[0][:], grid[1][:]

    for i in range(1, n):
        pre1[i] += pre1[i - 1]
        pre2[i] += pre2[i - 1]

    res = float("inf")
    for i in range(n):
        top = pre1[-1] - pre1[i]
        bottom = pre2[i - 1] if i > 0 else 0
        secondRobot = max(top, bottom)
        res = min(res, secondRobot)
    return res

Complexity: O(n) Time | O(n) Space

#438 FIND ALL ANAGRAMS

Find all start indices of p's anagrams in s.

def findAnagrams(s: str, p: str) -> list[int]:
    if len(p) > len(s): return []
    pCount, sCount = {}, {}
    for i in range(len(p)):
        pCount[p[i]] = 1 + pCount.get(p[i], 0)
        sCount[s[i]] = 1 + sCount.get(s[i], 0)

    res = [0] if sCount == pCount else []
    l = 0
    for r in range(len(p), len(s)):
        sCount[s[r]] = 1 + sCount.get(s[r], 0)
        sCount[s[l]] -= 1
        if sCount[s[l]] == 0:
            sCount.pop(s[l])
        l += 1
        if sCount == pCount:
            res.append(l)
    return res

Complexity: O(n) Time | O(1) Space (26 chars)

#560 SUBARRAY SUM K

Find the total number of continuous subarrays whose sum equals to k.

def subarraySum(nums: list[int], k: int) -> int:
    res, curSum = 0, 0
    prefixSums = {0: 1}

    for n in nums:
        curSum += n
        diff = curSum - k
        res += prefixSums.get(diff, 0)
        prefixSums[curSum] = 1 + prefixSums.get(curSum, 0)
    return res

Complexity: O(n) Time | O(n) Space

#554 BRICK WALL

Find the vertical line that crosses the fewest bricks.

def leastBricks(wall: list[list[int]]) -> int:
    countGap = {0: 0}
    for row in wall:
        total = 0
        for b in row[:-1]:
            total += b
            countGap[total] = 1 + countGap.get(total, 0)
    return len(wall) - max(countGap.values())

Complexity: O(n) Time | O(m) Space (where m is number of gaps)

#122 BUY/SELL STOCK II

Find the maximum profit you can achieve by buying and selling multiple times.

def maxProfit(prices: list[int]) -> int:
    profit = 0
    for i in range(1, len(prices)):
        if prices[i] > prices[i - 1]:
            profit += (prices[i] - prices[i - 1])
    return profit

Complexity: O(n) Time | O(1) Space

#167 TWO SUM II

Find two numbers that add up to target in a 1-indexed sorted array.

def twoSum(numbers: list[int], target: int) -> list[int]:
    l, r = 0, len(numbers) - 1
    while l < r:
        curSum = numbers[l] + numbers[r]
        if curSum > target:
            r -= 1
        elif curSum < target:
            l += 1
        else:
            return [l + 1, r + 1]

Complexity: O(n) Time | O(1) Space
