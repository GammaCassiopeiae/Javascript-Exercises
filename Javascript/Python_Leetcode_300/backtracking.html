#78 SUBSETS

Return all possible subsets (the power set) of a set of unique integers.

def subsets(nums: list[int]) -> list[list[int]]:
    res = []
    subset = []

    def dfs(i):
        if i >= len(nums):
            res.append(subset.copy())
            return
        
        # Decision to include nums[i]
        subset.append(nums[i])
        dfs(i + 1)
        
        # Decision NOT to include nums[i]
        subset.pop()
        dfs(i + 1)

    dfs(0)
    return res

Complexity: O(n * 2^n) Time | O(n) Space

#39 COMBINATION SUM

Find all unique combinations in candidates where the numbers sum to target.

def combinationSum(candidates: list[int], target: int) -> list[list[int]]:
    res = []

    def dfs(i, cur, total):
        if total == target:
            res.append(cur.copy())
            return
        if i >= len(candidates) or total > target:
            return

        cur.append(candidates[i])
        dfs(i, cur, total + candidates[i])
        cur.pop()
        dfs(i + 1, cur, total)

    dfs(0, [], 0)
    return res

Complexity: O(2^t) Time | O(t) Space (t = target value)

#46 PERMUTATIONS

Return all possible permutations of an array of distinct integers.

def permute(nums: list[int]) -> list[list[int]]:
    res = []
    if len(nums) == 1:
        return [nums.copy()]

    for i in range(len(nums)):
        n = nums.pop(0)
        perms = permute(nums)
        for p in perms:
            p.append(n)
        res.extend(perms)
        nums.append(n)
    return res

Complexity: O(n! * n) Time | O(n!) Space

#90 SUBSETS II

Return all possible subsets from a list that may contain duplicates.

def subsetsWithDup(nums: list[int]) -> list[list[int]]:
    res = []
    nums.sort()

    def backtrack(i, subset):
        if i == len(nums):
            res.append(subset[::])
            return

        # All subsets that include nums[i]
        subset.append(nums[i])
        backtrack(i + 1, subset)
        subset.pop()

        # All subsets that don't include nums[i]
        while i + 1 < len(nums) and nums[i] == nums[i + 1]:
            i += 1
        backtrack(i + 1, subset)

    backtrack(0, [])
    return res

Complexity: O(n * 2^n) Time | O(n) Space

#40 COMBINATION SUM II

Find all unique combinations where the sum equals target, each number used once.

def combinationSum2(candidates: list[int], target: int) -> list[list[int]]:
    candidates.sort()
    res = []

    def backtrack(cur, pos, target):
        if target == 0:
            res.append(cur.copy())
            return
        if target <= 0:
            return

        prev = -1
        for i in range(pos, len(candidates)):
            if candidates[i] == prev:
                continue
            cur.append(candidates[i])
            backtrack(cur, i + 1, target - candidates[i])
            cur.pop()
            prev = candidates[i]

    backtrack([], 0, target)
    return res

Complexity: O(2^n) Time | O(n) Space

#79 WORD SEARCH

Check if a word exists in a 2D grid of characters.

def exist(board: list[list[str]], word: str) -> bool:
    ROWS, COLS = len(board), len(board[0])
    path = set()

    def dfs(r, c, i):
        if i == len(word):
            return True
        if (r < 0 or c < 0 or r >= ROWS or c >= COLS or
            word[i] != board[r][c] or (r, c) in path):
            return False

        path.add((r, c))
        res = (dfs(r + 1, c, i + 1) or dfs(r - 1, c, i + 1) or
               dfs(r, c + 1, i + 1) or dfs(r, c - 1, i + 1))
        path.remove((r, c))
        return res

    for r in range(ROWS):
        for c in range(COLS):
            if dfs(r, c, 0): return True
    return False

Complexity: O(n * m * 4^n) Time | O(n) Space

#131 PALINDROME PARTITIONING

Partition a string such that every substring is a palindrome.

def partition(s: str) -> list[list[str]]:
    res = []
    part = []

    def dfs(i):
        if i >= len(s):
            res.append(part.copy())
            return
        for j in range(i, len(s)):
            if isPali(s, i, j):
                part.append(s[i : j + 1])
                dfs(j + 1)
                part.pop()

    def isPali(s, l, r):
        while l < r:
            if s[l] != s[r]: return False
            l, r = l + 1, r - 1
        return True

    dfs(0)
    return res

Complexity: O(2^n) Time | O(n) Space

#17 LETTER COMBINATIONS

Return all possible letter combinations that the number could represent.

def letterCombinations(digits: str) -> list[str]:
    res = []
    digitToChar = {
        "2": "abc", "3": "def", "4": "ghi", "5": "jkl",
        "6": "mno", "7": "pqrs", "8": "tuv", "9": "wxyz",
    }

    def backtrack(i, curStr):
        if len(curStr) == len(digits):
            res.append(curStr)
            return
        for c in digitToChar[digits[i]]:
            backtrack(i + 1, curStr + c)

    if digits: backtrack(0, "")
    return res

Complexity: O(n * 4^n) Time | O(n) Space

#93 RESTORE IP ADDRESSES

Return all possible valid IP addresses that can be formed from s.

def restoreIpAddresses(s: str) -> list[str]:
    res = []
    if len(s) > 12: return res

    def backtrack(i, dots, curIP):
        if dots == 4 and i == len(s):
            res.append(curIP[:-1])
            return
        if dots > 4: return

        for j in range(i, min(i + 3, len(s))):
            if int(s[i : j + 1]) < 256 and (i == j or s[i] != "0"):
                backtrack(j + 1, dots + 1, curIP + s[i : j + 1] + ".")

    backtrack(0, 0, "")
    return res

Complexity: O(3^4) Time | O(1) Space

#473 MATCHSTICKS TO SQUARE

Determine if you can form a square using all the given matchsticks.

def makesquare(matchsticks: list[int]) -> bool:
    length = sum(matchsticks) // 4
    sides = [0] * 4
    if sum(matchsticks) / 4 != length: return False
    matchsticks.sort(reverse=True)

    def backtrack(i):
        if i == len(matchsticks): return True
        for j in range(4):
            if sides[j] + matchsticks[i] <= length:
                sides[j] += matchsticks[i]
                if backtrack(i + 1): return True
                sides[j] -= matchsticks[i]
            if sides[j] == 0: break
        return False

    return backtrack(0)

Complexity: O(4^n) Time | O(n) Space

#1849 SPLIT STRING DESCENDING

Check if s can be split into descending consecutive values.

def splitString(s: str) -> bool:
    def backtrack(index, prev):
        if index == len(s): return True
        for j in range(index, len(s)):
            val = int(s[index : j + 1])
            if val + 1 == prev and backtrack(j + 1, val):
                return True
        return False

    for i in range(len(s) - 1):
        val = int(s[: i + 1])
        if backtrack(i + 1, val): return True
    return False

Complexity: O(n^n) Time | O(n) Space

#51 N-QUEENS

Place n queens on an n×n chessboard such that no two queens attack each other.

def solveNQueens(n: int) -> list[list[str]]:
    col, posDiag, negDiag = set(), set(), set()
    res = []
    board = [["."] * n for _ in range(n)]

    def backtrack(r):
        if r == n:
            copy = ["".join(row) for row in board]
            res.append(copy)
            return

        for c in range(n):
            if c in col or (r + c) in posDiag or (r - c) in negDiag:
                continue

            col.add(c); posDiag.add(r + c); negDiag.add(r - c)
            board[r][c] = "Q"
            backtrack(r + 1)
            col.remove(c); posDiag.remove(r + c); negDiag.remove(r - c)
            board[r][c] = "."

    backtrack(0)
    return res

Complexity: O(n!) Time | O(n^2) Space

#212 WORD SEARCH II

Find all words in the board that are also in the given list of words.

class TrieNode:
    def __init__(self):
        self.children = {}; self.isWord = False
    def addWord(self, word):
        cur = self
        for c in word:
            if c not in cur.children: cur.children[c] = TrieNode()
            cur = cur.children[c]
        cur.isWord = True

def findWords(board, words):
    root = TrieNode()
    for w in words: root.addWord(w)
    ROWS, COLS = len(board), len(board[0])
    res, visit = set(), set()

    def dfs(r, c, node, word):
        if (r < 0 or c < 0 or r == ROWS or c == COLS or
            (r, c) in visit or board[r][c] not in node.children):
            return
        visit.add((r, c))
        node = node.children[board[r][c]]
        word += board[r][c]
        if node.isWord: res.add(word)
        dfs(r + 1, c, node, word); dfs(r - 1, c, node, word)
        dfs(r, c + 1, node, word); dfs(r, c - 1, node, word)
        visit.remove((r, c))

    for r in range(ROWS):
        for c in range(COLS): dfs(r, c, root, "")
    return list(res)

Complexity: O(M * (4 * 3^(L-1))) Time | O(N) Space (M=cells, L=word len, N=total chars in Trie)

#47 PERMUTATIONS II

Return all possible unique permutations of a collection that might contain duplicates.

def permuteUnique(nums: list[int]) -> list[list[int]]:
    res = []
    perm = []
    count = {n: 0 for n in nums}
    for n in nums: count[n] += 1

    def dfs():
        if len(perm) == len(nums):
            res.append(perm.copy())
            return
        for n in count:
            if count[n] > 0:
                perm.append(n); count[n] -= 1
                dfs()
                count[n] += 1; perm.pop()

    dfs()
    return res

Complexity: O(n! * n) Time | O(n) Space

#216 COMBINATION SUM III

Find all valid combinations of k numbers that sum up to n using numbers 1-9.

def combinationSum3(k: int, n: int) -> list[list[int]]:
    res = []

    def backtrack(num, stack, target):
        if len(stack) == k:
            if target == 0: res.append(stack[:])
            return
        for x in range(num, 10):
            if x > target: break
            stack.append(x)
            backtrack(x + 1, stack, target - x)
            stack.pop()

    backtrack(1, [], n)
    return res

Complexity: O(C(9, k)) Time | O(k) Space

#37 SUDOKU SOLVER

Write a program to solve a Sudoku puzzle by filling the empty cells.

def solveSudoku(board: list[list[str]]) -> None:
    def isValid(r, c, char):
        for i in range(9):
            if board[i][c] == char: return False
            if board[r][i] == char: return False
            if board[3 * (r // 3) + i // 3][3 * (c // 3) + i % 3] == char: return False
        return True

    def solve():
        for r in range(9):
            for c in range(9):
                if board[r][c] == ".":
                    for char in "123456789":
                        if isValid(r, c, char):
                            board[r][c] = char
                            if solve(): return True
                            board[r][c] = "."
                    return False
        return True
    solve()

Complexity: O(9^(n*n)) Time | O(n*n) Space

#22 GENERATE PARENTHESES

Generate all combinations of well-formed parentheses given n pairs.

def generateParenthesis(n: int) -> list[str]:
    stack = []
    res = []

    def backtrack(openN, closedN):
        if openN == closedN == n:
            res.append("".join(stack))
            return
        if openN < n:
            stack.append("(")
            backtrack(openN + 1, closedN)
            stack.pop()
        if closedN < openN:
            stack.append(")")
            backtrack(openN, closedN + 1)
            stack.pop()

    backtrack(0, 0)
    return res

Complexity: O(4^n / sqrt(n)) Time | O(n) Space

#526 BEAUTIFUL ARRANGEMENT

Find the number of beautiful arrangements that can be constructed.

def countArrangement(n: int) -> int:
    visited = [False] * (n + 1)
    count = 0

    def backtrack(pos):
        nonlocal count
        if pos > n:
            count += 1
            return
        for i in range(1, n + 1):
            if not visited[i] and (i % pos == 0 or pos % i == 0):
                visited[i] = True
                backtrack(pos + 1)
                visited[i] = False

    backtrack(1)
    return count

Complexity: O(k) Time (k is number of beautiful arrangements) | O(n) Space

#698 PARTITION K SUBSETS

Check if it's possible to partition nums into k subsets with equal sum.

def canPartitionKSubsets(nums: list[int], k: int) -> bool:
    if sum(nums) % k != 0: return False
    target = sum(nums) / k
    used = [False] * len(nums)
    nums.sort(reverse=True)

    def backtrack(i, k, subsetSum):
        if k == 0: return True
        if subsetSum == target:
            return backtrack(0, k - 1, 0)
        for j in range(i, len(nums)):
            if used[j] or subsetSum + nums[j] > target:
                continue
            used[j] = True
            if backtrack(j + 1, k, subsetSum + nums[j]):
                return True
            used[j] = False
            if subsetSum == 0: break
        return False

    return backtrack(0, k, 0)

Complexity: O(k * 2^n) Time | O(n) Space

#980 UNIQUE PATHS III

Return the number of 4-directional walks from start to end that visit every non-obstacle square exactly once.

def uniquePathsIII(grid: list[list[int]]) -> int:
    ROWS, COLS = len(grid), len(grid[0])
    start_r, start_c, empty = 0, 0, 1
    for r in range(ROWS):
        for c in range(COLS):
            if grid[r][c] == 1: start_r, start_c = r, c
            elif grid[r][c] == 0: empty += 1

    res = 0
    def backtrack(r, c, visited_count):
        nonlocal res
        if not (0 <= r < ROWS and 0 <= c < COLS) or grid[r][c] == -1:
            return
        if grid[r][c] == 2:
            if visited_count == empty: res += 1
            return

        temp = grid[r][c]
        grid[r][c] = -1
        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            backtrack(r + dr, c + dc, visited_count + 1)
        grid[r][c] = temp

    backtrack(start_r, start_c, 0)
    return res

Complexity: O(3^n) Time | O(n) Space
