#704 BINARY SEARCH

The fundamental algorithm for finding a target in a sorted array.

def search(nums: list[int], target: int) -> int:
    l, r = 0, len(nums) - 1
    while l <= r:
        m = l + ((r - l) // 2)
        if nums[m] > target:
            r = m - 1
        elif nums[m] < target:
            l = m + 1
        else:
            return m
    return -1

Complexity: O(log n) Time | O(1) Space

#35 SEARCH INSERT POSITION

Return the index if the target is found. If not, return the index where it would be if it were inserted in order.

def searchInsert(nums: list[int], target: int) -> int:
    l, r = 0, len(nums) - 1
    while l <= r:
        mid = (l + r) // 2
        if nums[mid] == target:
            return mid
        if nums[mid] < target:
            l = mid + 1
        else:
            r = mid - 1
    return l

Complexity: O(log n) Time | O(1) Space

#441 ARRANGING COINS

Find the total number of full rows of staircase that can be formed from n coins.

def arrangeCoins(n: int) -> int:
    l, r = 1, n
    res = 0
    while l <= r:
        mid = (l + r) // 2
        coins = (mid / 2) * (mid + 1)
        if coins > n:
            r = mid - 1
        else:
            l = mid + 1
            res = max(res, mid)
    return res

Complexity: O(log n) Time | O(1) Space

#367 VALID PERFECT SQUARE

Determine if a positive integer is a perfect square without using built-in functions.

def isPerfectSquare(num: int) -> bool:
    l, r = 1, num
    while l <= r:
        mid = (l + r) // 2
        if mid * mid > num:
            r = mid - 1
        elif mid * mid < num:
            l = mid + 1
        else:
            return True
    return False

Complexity: O(log n) Time | O(1) Space

#74 SEARCH A 2D MATRIX

Search for a target value in an m x n matrix with sorted rows and columns.

def searchMatrix(matrix: list[list[int]], target: int) -> bool:
    ROWS, COLS = len(matrix), len(matrix[0])
    top, bot = 0, ROWS - 1
    while top <= bot:
        row = (top + bot) // 2
        if target > matrix[row][-1]:
            top = row + 1
        elif target < matrix[row][0]:
            bot = row - 1
        else:
            break
    if not (top <= bot): return False
    row = (top + bot) // 2
    l, r = 0, COLS - 1
    while l <= r:
        m = (l + r) // 2
        if target > matrix[row][m]:
            l = m + 1
        elif target < matrix[row][m]:
            r = m - 1
        else:
            return True
    return False

Complexity: O(log m + log n) Time | O(1) Space

#875 KOKO EATING BANANAS

Find the minimum integer k such that Koko can eat all bananas within h hours.

import math

def minEatingSpeed(piles: list[int], h: int) -> int:
    l, r = 1, max(piles)
    res = r
    while l <= r:
        k = (l + r) // 2
        hours = 0
        for p in piles:
            hours += math.ceil(p / k)
        if hours <= h:
            res = k
            r = k - 1
        else:
            l = k + 1
    return res

Complexity: O(n log(max(piles))) Time | O(1) Space

#153 FIND MINIMUM IN ROTATED SORTED ARRAY

Find the minimum element in a sorted array that has been rotated.

def findMin(nums: list[int]) -> int:
    res = nums[0]
    l, r = 0, len(nums) - 1
    while l <= r:
        if nums[l] < nums[r]:
            res = min(res, nums[l])
            break
        m = (l + r) // 2
        res = min(res, nums[m])
        if nums[m] >= nums[l]:
            l = m + 1
        else:
            r = m - 1
    return res

Complexity: O(log n) Time | O(1) Space

#33 SEARCH IN ROTATED SORTED ARRAY

Find a target in a sorted array that has been rotated.

def search(nums: list[int], target: int) -> int:
    l, r = 0, len(nums) - 1
    while l <= r:
        mid = (l + r) // 2
        if target == nums[mid]: return mid
        if nums[l] <= nums[mid]:
            if target > nums[mid] or target < nums[l]:
                l = mid + 1
            else:
                r = mid - 1
        else:
            if target < nums[mid] or target > nums[r]:
                r = mid - 1
            else:
                l = mid + 1
    return -1

Complexity: O(log n) Time | O(1) Space

#34 FIND FIRST AND LAST POSITION OF ELEMENT

Find the starting and ending position of a given target value in a sorted array.

def searchRange(nums: list[int], target: int) -> list[int]:
    def binSearch(leftBias):
        l, r = 0, len(nums) - 1
        i = -1
        while l <= r:
            m = (l + r) // 2
            if target > nums[m]: l = m + 1
            elif target < nums[m]: r = m - 1
            else:
                i = m
                if leftBias: r = m - 1
                else: l = m + 1
        return i
    return [binSearch(True), binSearch(False)]

Complexity: O(log n) Time | O(1) Space

#540 SINGLE ELEMENT IN A SORTED ARRAY

Find the only element that appears exactly once in an array where every other element appears twice.

def singleNonDuplicate(nums: list[int]) -> int:
    l, r = 0, len(nums) - 1
    while l < r:
        m = (l + r) // 2
        if (m % 2 == 1 and nums[m] == nums[m - 1]) or (m % 2 == 0 and nums[m] == nums[m + 1]):
            l = m + 1
        else:
            r = m
    return nums[l]

Complexity: O(log n) Time | O(1) Space

#1011 CAPACITY TO SHIP PACKAGES WITHIN D DAYS

Find the minimum weight capacity of a ship to deliver all packages within d days.

def shipWithinDays(weights: list[int], days: int) -> int:
    l, r = max(weights), sum(weights)
    res = r
    def canShip(cap):
        ships, curCap = 1, cap
        for w in weights:
            if curCap - w < 0:
                ships += 1
                curCap = cap
            curCap -= w
        return ships <= days
    while l <= r:
        cap = (l + r) // 2
        if canShip(cap):
            res = cap
            r = cap - 1
        else:
            l = cap + 1
    return res

Complexity: O(n log(sum-max)) Time | O(1) Space

#2300 SUCCESSFUL PAIRS OF SPELLS AND POTIONS

For each spell, find the number of potions that form a successful pair.

def successfulPairs(spells: list[int], potions: list[int], success: int) -> list[int]:
    potions.sort()
    res = []
    for s in spells:
        l, r = 0, len(potions) - 1
        idx = len(potions)
        while l <= r:
            m = (l + r) // 2
            if s * potions[m] >= success:
                r = m - 1
                idx = m
            else:
                l = m + 1
        res.append(len(potions) - idx)
    return res

Complexity: O(n log n + m log n) Time | O(1) Space

#4 MEDIAN OF TWO SORTED ARRAYS

Find the median of two sorted arrays in logarithmic time.

def findMedianSortedArrays(nums1: list[int], nums2: list[int]) -> float:
    A, B = nums1, nums2
    total = len(nums1) + len(nums2)
    half = total // 2
    if len(B) < len(A): A, B = B, A
    l, r = 0, len(A) - 1
    while True:
        i = (l + r) // 2
        j = half - i - 2
        Aleft = A[i] if i >= 0 else float("-infinity")
        Aright = A[i + 1] if (i + 1) < len(A) else float("infinity")
        Bleft = B[j] if j >= 0 else float("-infinity")
        Bright = B[j + 1] if (j + 1) < len(B) else float("infinity")
        if Aleft <= Bright and Bleft <= Aright:
            if total % 2: return min(Aright, Bright)
            return (max(Aleft, Bleft) + min(Aright, Bright)) / 2
        elif Aleft > Bright: r = i - 1
        else: l = i + 1

Complexity: O(log(min(n, m))) Time | O(1) Space

#410 SPLIT ARRAY LARGEST SUM

Split the array into k non-empty subarrays such that the largest sum of any subarray is minimized.

def splitArray(nums: list[int], k: int) -> int:
    def canSplit(maxSum):
        subarray, curSum = 1, 0
        for n in nums:
            curSum += n
            if curSum > maxSum:
                subarray += 1
                curSum = n
        return subarray <= k
    l, r = max(nums), sum(nums)
    res = r
    while l <= r:
        mid = l + ((r - l) // 2)
        if canSplit(mid):
            res = mid
            r = mid - 1
        else:
            l = mid + 1
    return res

Complexity: O(n log(sum-max)) Time | O(1) Space
