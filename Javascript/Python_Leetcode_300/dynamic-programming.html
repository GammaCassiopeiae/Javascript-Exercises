#70 CLIMBING STAIRS

Find the number of ways to climb n stairs where you can take 1 or 2 steps at a time.

def climbStairs(n: int) -> int:
    if n <= 2: return n
    one, two = 1, 1
    for i in range(n - 1):
        temp = one
        one = one + two
        two = temp
    return one

Complexity: O(n) Time | O(1) Space

#746 MIN COST CLIMBING STAIRS

Find the minimum cost to reach the top of the floor.

def minCostClimbingStairs(cost: list[int]) -> int:
    cost.append(0)
    for i in range(len(cost) - 3, -1, -1):
        cost[i] += min(cost[i + 1], cost[i + 2])
    return min(cost[0], cost[1])

Complexity: O(n) Time | O(1) Space

#198 HOUSE ROBBER

Maximize loot from houses without robbing two adjacent ones.

def rob(nums: list[int]) -> int:
    rob1, rob2 = 0, 0
    for n in nums:
        temp = max(n + rob1, rob2)
        rob1 = rob2
        rob2 = temp
    return rob2

Complexity: O(n) Time | O(1) Space

Houses are arranged in a circle. Maximize loot without robbing adjacent ones.

def rob(nums: list[int]) -> int:
    def helper(n):
        r1, r2 = 0, 0
        for x in n:
            r1, r2 = r2, max(x + r1, r2)
        return r2
    return max(nums[0], helper(nums[1:]), helper(nums[:-1]))

Complexity: O(n) Time | O(1) Space

#5 LONGEST PALINDROMIC SUBSTRING

Find the longest substring that is a palindrome.

def longestPalindrome(s: str) -> str:
    res = ""
    for i in range(len(s)):
        # Odd length
        l, r = i, i
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if (r - l + 1) > len(res): res = s[l:r+1]
            l -= 1; r += 1
        # Even length
        l, r = i, i + 1
        while l >= 0 and r < len(s) and s[l] == s[r]:
            if (r - l + 1) > len(res): res = s[l:r+1]
            l -= 1; r += 1
    return res

Complexity: O(n²) Time | O(1) Space

def countSubstrings(s: str) -> int:

#91 DECODE WAYS

Find total ways to decode a numeric string (A=1, B=2...).

def numDecodings(s: str) -> int:
    dp = {len(s): 1}
    for i in range(len(s) - 1, -1, -1):
        if s[i] == "0": dp[i] = 0
        else: dp[i] = dp[i + 1]
        if i + 1 < len(s) and (s[i] == "1" or (s[i] == "2" and s[i + 1] in "0123456")):
            dp[i] += dp[i + 2]
    return dp[0]

Complexity: O(n) Time | O(n) Space

#322 COIN CHANGE

Minimum number of coins to make up a target amount.

def coinChange(coins: list[int], amount: int) -> int:
    dp = [amount + 1] * (amount + 1)
    dp[0] = 0
    for a in range(1, amount + 1):
        for c in coins:
            if a - c >= 0:
                dp[a] = min(dp[a], 1 + dp[a - c])
    return dp[amount] if dp[amount] != amount + 1 else -1

Complexity: O(amount * n) Time | O(amount) Space

#152 MAXIMUM PRODUCT SUBARRAY

Find the contiguous subarray with the largest product.

def maxProduct(nums: list[int]) -> int:
    res = max(nums)
    curMin, curMax = 1, 1
    for n in nums:
        if n == 0:
            curMin, curMax = 1, 1
            continue
        tmp = curMax * n
        curMax = max(n * curMax, n * curMin, n)
        curMin = min(tmp, n * curMin, n)
        res = max(res, curMax)
    return res

Complexity: O(n) Time | O(1) Space

#300 LONGEST INCREASING SUBSEQUENCE

Find the length of the longest strictly increasing subsequence.

def lengthOfLIS(nums: list[int]) -> int:

    for i in range(len(nums) - 1, -1, -1):
        for j in range(i + 1, len(nums)):

#139 WORD BREAK

Determine if a string can be segmented into space-separated dictionary words.

def wordBreak(s: str, wordDict: list[str]) -> bool:
    dp = [False] * (len(s) + 1)
    dp[len(s)] = True
    for i in range(len(s) - 1, -1, -1):
        for w in wordDict:
            if (i + len(w)) <= len(s) and s[i : i + len(w)] == w:
                dp[i] = dp[i + len(w)]
            if dp[i]: break
    return dp[0]

Complexity: O(n * m * k) Time | O(n) Space

#416 PARTITION EQUAL SUBSET SUM

Determine if an array can be partitioned into two subsets with equal sum.

def canPartition(nums: list[int]) -> bool:
    if sum(nums) % 2: return False
    target = sum(nums) // 2
    dp = set([0])
    for n in nums:
        nextDP = set()
        for t in dp:
            if t + n == target: return True
            nextDP.add(t + n)
            nextDP.add(t)
        dp = nextDP
    return False

Complexity: O(n * sum) Time | O(sum) Space

#740 DELETE AND EARN

Maximize points by deleting numbers and their neighbors.

def deleteAndEarn(nums: list[int]) -> int:
    count = collections.Counter(nums)
    nums = sorted(list(set(nums)))
    rob1, rob2 = 0, 0
    for i in range(len(nums)):
        curVal = nums[i] * count[nums[i]]
        if i > 0 and nums[i] == nums[i - 1] + 1:
            temp = max(curVal + rob1, rob2)
            rob1 = rob2
            rob2 = temp
        else:
            temp = curVal + rob2
            rob1 = rob2
            rob2 = temp
    return rob2

Complexity: O(n log n) Time | O(n) Space

#32 LONGEST VALID PARENTHESES

Find the length of the longest valid (well-formed) parentheses substring.

def longestValidParentheses(s: str) -> int:
    stack = [-1]
    res = 0
    for i, char in enumerate(s):
        if char == '(':
            stack.append(i)
        else:
            stack.pop()
            if not stack:
                stack.append(i)
            else:
                res = max(res, i - stack[-1])
    return res

Complexity: O(n) Time | O(n) Space

#983 MINIMUM COST FOR TICKETS

Find minimum dollars needed to travel every day in the given list of days.

def mincostTickets(days: list[int], costs: list[int]) -> int:
    dp = [0] * (days[-1] + 1)
    day_set = set(days)
    for i in range(1, len(dp)):
        if i not in day_set:
            dp[i] = dp[i-1]
        else:
            dp[i] = min(dp[i-1] + costs[0],
                        dp[max(0, i-7)] + costs[1],
                        dp[max(0, i-30)] + costs[2])
    return dp[-1]

Complexity: O(max_day) Time | O(max_day) Space

#279 PERFECT SQUARES

Find the least number of perfect square numbers that sum to n.

def numSquares(n: int) -> int:
    dp = [n] * (n + 1)
    dp[0] = 0
    for target in range(1, n + 1):
        for s in range(1, target + 1):
            square = s * s
            if target - square < 0:
                break
            dp[target] = min(dp[target], 1 + dp[target - square])
    return dp[n]

Complexity: O(n * sqrt(n)) Time | O(n) Space

#1043 PARTITION ARRAY FOR MAX SUM

Partition array into subarrays of length at most k to maximize sum.

def maxSumAfterPartitioning(arr: list[int], k: int) -> int:
    dp = [0] * (len(arr) + 1)
    for i in range(1, len(arr) + 1):
        curMax = 0
        for j in range(1, min(k, i) + 1):
            curMax = max(curMax, arr[i - j])
            dp[i] = max(dp[i], dp[i - j] + curMax * j)
    return dp[len(arr)]

Complexity: O(n * k) Time | O(n) Space

#673 NUMBER OF LIS

Find the number of longest increasing subsequences.

def findNumberOfLIS(nums: list[int]) -> int:
    dp = {} # index -> [length of LIS, count]
    lenLIS, res = 0, 0
    for i in range(len(nums) - 1, -1, -1):
        curLen, curCnt = 1, 1
        for j in range(i + 1, len(nums)):
            if nums[j] > nums[i]:
                l, c = dp[j]
                if l + 1 > curLen:
                    curLen, curCnt = l + 1, c
                elif l + 1 == curLen:
                    curCnt += c
        dp[i] = [curLen, curCnt]
        if curLen > lenLIS:
            lenLIS, res = curLen, curCnt
        elif curLen == lenLIS:
            res += curCnt
    return res

Complexity: O(n²) Time | O(n) Space

#1186 MAX SUBARRAY SUM WITH ONE DELETION

Find max sum of non-empty subarray with at most one deletion.

def maximumSum(arr: list[int]) -> int:
    n = len(arr)
    one_del = [0] * n
    no_del = [0] * n
    no_del[0] = arr[0]
    one_del[0] = -float('inf')
    res = arr[0]
    for i in range(1, n):
        no_del[i] = max(arr[i], no_del[i-1] + arr[i])
        one_del[i] = max(no_del[i-1], one_del[i-1] + arr[i])
        res = max(res, no_del[i], one_del[i])
    return res

Complexity: O(n) Time | O(n) Space
