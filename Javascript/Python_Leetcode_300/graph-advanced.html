#200 NUMBER OF ISLANDS

Count the number of islands in a 2D binary grid.

def numIslands(grid: list[list[str]]) -> int:
    if not grid: return 0
    ROWS, COLS = len(grid), len(grid[0])
    islands = 0
    def dfs(r, c):
        if r < 0 or c < 0 or r == ROWS or c == COLS or grid[r][c] == "0":
            return
        grid[r][c] = "0"
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            dfs(r + dr, c + dc)
    for r in range(ROWS):
        for c in range(COLS):
            if grid[r][c] == "1":
                islands += 1
                dfs(r, c)
    return islands

Complexity: O(M * N) Time | O(M * N) Space

def maxAreaOfIsland(grid: list[list[int]]) -> int:

    visit = set()
    def dfs(r, c):
        if (r < 0 or r == ROWS or c < 0 or c == COLS or 
            grid[r][c] == 0 or (r, c) in visit):
            return 0
        visit.add((r, c))
        return 1 + dfs(r+1, c) + dfs(r-1, c) + dfs(r, c+1) + dfs(r, c-1)

        for c in range(COLS):
            area = max(area, dfs(r, c))
    return area

def pacificAtlantic(heights: list[list[int]]) -> list[list[int]]:
    ROWS, COLS = len(heights), len(heights[0])
    pac, atl = set(), set()
    def dfs(r, c, visit, prevHeight):
        if ((r, c) in visit or r < 0 or c < 0 or r == ROWS or c == COLS or 
            heights[r][c] < prevHeight): return

#323 CONNECTED COMPONENTS

Find the number of connected components in an undirected graph.

def countComponents(n, edges):
    par = [i for i in range(n)]; rank = [1] * n
    def find(n1):
        res = n1
        while res != par[res]:
            par[res] = par[par[res]]; res = par[res]
        return res
    def union(n1, n2):
        p1, p2 = find(n1), find(n2)
        if p1 == p2: return 0
        if rank[p1] > rank[p2]: par[p2] = p1; rank[p1] += rank[p2]
        else: par[p1] = p2; rank[p2] += rank[p1]
        return 1
    res = n
    for n1, n2 in edges: res -= union(n1, n2)
    return res

Complexity: O(E * α(V)) Time | O(V) Space

#743 NETWORK DELAY TIME

import heapq
def networkDelayTime(times, n, k):
    adj = collections.defaultdict(list)
    for u, v, w in times: adj[u].append((v, w))
    min_heap = [(0, k)]; visit = set(); t = 0
    while min_heap:
        w1, n1 = heapq.heappop(min_heap)
        if n1 in visit: continue
        visit.add(n1); t = max(t, w1)
        for n2, w2 in adj[n1]:
            if n2 not in visit: heapq.heappush(min_heap, (w1 + w2, n2))
    return t if len(visit) == n else -1

# DIJKSTRA'S ALGORITHM

Standard shortest path for non-negative weights.

def dijkstra(graph, start):
    dist = {node: float('inf') for node in graph}; dist[start] = 0
    pq = [(0, start)]
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: continue
        for v, w in graph[u].items():
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w; heapq.heappush(pq, (dist[v], v))
    return dist

Complexity: O(E log V) Time | O(V) Space

# BELLMAN-FORD ALGORITHM

Shortest path with possible negative edge weights.

def bellman_ford(V, edges, start):
    dist = [float('inf')] * V; dist[start] = 0
    for _ in range(V - 1):
        for u, v, w in edges:
            if dist[u] != float('inf') and dist[u] + w < dist[v]: dist[v] = dist[u] + w
    return dist

Complexity: O(V * E) Time | O(V) Space

# FLOYD-WARSHALL ALGORITHM

All-pairs shortest path algorithm.

def floyd_warshall(graph, V):
    dist = [row[:] for row in graph]
    for k in range(V):
        for i in range(V):
            for j in range(V):
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])
    return dist

Complexity: O(V³) Time | O(V²) Space

# KRUSKAL'S ALGORITHM

MST using Union-Find.

def kruskal(n, edges):
    edges.sort(key=lambda x: x[2])
    par = list(range(n))
    def find(i):
        if par[i] == i: return i
        par[i] = find(par[i]); return par[i]
    mst = []; cost = 0
    for u, v, w in edges:
        root_u, root_v = find(u), find(v)
        if root_u != root_v:
            par[root_u] = root_v; mst.append((u, v, w)); cost += w
    return mst, cost

Complexity: O(E log E) Time | O(V) Space

def isBipartite(graph: list[list[int]]) -> bool:
    color = {} # node: 0/1

        if i not in color:

                node = q.popleft()

def canVisitAllRooms(rooms):
    visit = {0}; q = deque([0])
    while q:
        i = q.popleft()
        for key in rooms[i]:
            if key not in visit: visit.add(key); q.append(key)
    return len(visit) == len(rooms)

Complexity: O(V + E) Time | O(V) Space

#1020 NUMBER OF ENCLAVES

Count land cells that cannot reach the boundary.

def numEnclaves(grid):
    ROWS, COLS = len(grid), len(grid[0])
    def dfs(r, c):
        if r < 0 or c < 0 or r == ROWS or c == COLS or grid[r][c] == 0: return
        grid[r][c] = 0; [dfs(r+dr, c+dc) for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]]
    for r in range(ROWS):
        for c in [0, COLS-1]: dfs(r, c)
    for c in range(COLS):
        for r in [0, ROWS-1]: dfs(r, c)
    return sum(sum(row) for row in grid)

Complexity: O(M * N) Time | O(M * N) Space

#886 POSSIBLE BIPARTITION

Check if people can be split into two groups with no dislikes.

def possibleBipartition(n, dislikes):
    adj = collections.defaultdict(list)
    for u, v in dislikes: adj[u].append(v); adj[v].append(u)
    color = {}
    def dfs(i, c):
        if i in color: return color[i] == c
        color[i] = c
        return all(dfs(nei, 1 - c) for nei in adj[i])
    return all(dfs(i, 0) for i in range(1, n + 1) if i not in color)

Complexity: O(V + E) Time | O(V) Space

#399 EVALUATE DIVISION

Evaluate path values in a directed weighted graph.

def calcEquation(equations, values, queries):
    adj = collections.defaultdict(dict)
    for (a, b), val in zip(equations, values):
        adj[a][b] = val; adj[b][a] = 1 / val
    def bfs(src, dst):
        if src not in adj or dst not in adj: return -1.0
        q, visit = deque([(src, 1.0)]), {src}
        while q:
            n, v = q.popleft()
            if n == dst: return v
            for nei, val in adj[n].items():
                if nei not in visit: visit.add(nei); q.append((nei, v * val))
        return -1.0
    return [bfs(q[0], q[1]) for q in queries]

Complexity: O(Q * (V + E)) Time | O(V + E) Space

#329 LONGEST INCREASING PATH

Find the longest increasing path in a matrix.

def longestIncreasingPath(matrix):
    ROWS, COLS = len(matrix), len(matrix[0]); dp = {}
    def dfs(r, c, prev):
        if r < 0 or r == ROWS or c < 0 or c == COLS or matrix[r][c] <= prev: return 0
        if (r, c) in dp: return dp[(r, c)]
        res = 1 + max(dfs(r+1,c,matrix[r][c]), dfs(r-1,c,matrix[r][c]),
                      dfs(r,c+1,matrix[r][c]), dfs(r,c-1,matrix[r][c]))
        dp[(r, c)] = res; return res
    for r in range(ROWS):
        for c in range(COLS): dfs(r, c, -1)
    return max(dp.values())

Complexity: O(M * N) Time | O(M * N) Space

#2492 MIN SCORE PATH

Find the minimum edge weight in any path between node 1 and n.

def minScore(n, roads):
    adj = collections.defaultdict(list)
    for u, v, w in roads: adj[u].append((v, w)); adj[v].append((u, w))
    res = float("inf"); q = deque([1]); visit = {1}
    while q:
        node = q.popleft()
        for nei, w in adj[node]:
            res = min(res, w)
            if nei not in visit: visit.add(nei); q.append(nei)
    return res

Complexity: O(V + E) Time | O(V) Space

#2359 CLOSEST NODE TO TWO

Find a node with minimum distance to both given nodes.

def closestMeetingNode(edges, node1, node2):
    def getDist(node):
        dist = {node: 0}; curr = node
        while edges[curr] != -1 and edges[curr] not in dist:
            dist[edges[curr]] = dist[curr] + 1; curr = edges[curr]
        return dist
    d1, d2 = getDist(node1), getDist(node2)
    res, min_dist = -1, float("inf")
    for i in range(len(edges)):
        if i in d1 and i in d2:
            d = max(d1[i], d2[i])
            if d < min_dist: min_dist, res = d, i
    return res

Complexity: O(V) Time | O(V) Space

#934 SHORTEST BRIDGE

Find the minimum number of zeros to flip to connect two islands.

def shortestBridge(grid):
    n = len(grid); visit = set()
    def dfs(r, c):
        if r < 0 or c < 0 or r == n or c == n or grid[r][c] == 0 or (r, c) in visit: return
        visit.add((r, c)); [dfs(r+dr, c+dc) for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]]
    def find_first():
        for r in range(n):
            for c in range(n):
                if grid[r][c]: return r, c
    dfs(*find_first()); q = deque(list(visit)); d = 0
    while q:
        for _ in range(len(q)):
            r, c = q.popleft()
            for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < n and 0 <= nc < n and (nr, nc) not in visit:
                    if grid[nr][nc]: return d
                    visit.add((nr, nc)); q.append((nr, nc))
        d += 1

Complexity: O(N²) Time | O(N²) Space

#909 SNAKES AND LADDERS

Minimum moves to reach the last square on a board.

def snakesAndLadders(board):
    n = len(board); board.reverse()
    def getPos(pos):
        r = (pos - 1) // n; c = (pos - 1) % n
        if r % 2: c = n - 1 - c
        return r, c
    q = deque([(1, 0)]); visit = {1}
    while q:
        curr, moves = q.popleft()
        for i in range(1, 7):
            nxt = curr + i
            r, c = getPos(nxt)
            if board[r][c] != -1: nxt = board[r][c]
            if nxt == n * n: return moves + 1
            if nxt not in visit: visit.add(nxt); q.append((nxt, moves + 1))
    return -1

Complexity: O(N²) Time | O(N²) Space

# PRIM'S ALGORITHM

MST algorithm using a priority queue.

def prim(n, adj):
    visit = set(); minH = [[0, 0]]; res = 0
    while len(visit) < n:
        cost, i = heapq.heappop(minH)
        if i in visit: continue
        res += cost; visit.add(i)
        for nei, neiCost in adj[i]:
            if nei not in visit: heapq.heappush(minH, [neiCost, nei])
    return res

Complexity: O(E log V) Time | O(V + E) Space

#685 REDUNDANT CONNECTION II

Find an edge that can be removed to make a directed graph a rooted tree.

def findRedundantDirectedConnection(edges):
    n = len(edges); parent = [0] * (n + 1); cand1 = cand2 = None
    for u, v in edges:
        if parent[v]: cand1, cand2 = [parent[v], v], [u, v]; break
        parent[v] = u
    def find(i, p):
        if p[i] == i: return i
        p[i] = find(p[i], p); return p[i]
    p = list(range(n + 1))
    for u, v in edges:
        if [u, v] == cand2: continue
        root_u, root_v = find(u, p), find(v, p)
        if root_u == root_v: return cand1 if cand1 else [u, v]
        p[root_u] = root_v
    return cand2

Complexity: O(N * α(N)) Time | O(N) Space
