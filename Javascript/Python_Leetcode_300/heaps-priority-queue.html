#703 KTH LARGEST IN A STREAM

Design a class to find the kth largest element in a stream.

import heapq

class KthLargest:
    def __init__(self, k: int, nums: list[int]):
        self.k = k
        self.min_heap = nums
        heapq.heapify(self.min_heap)
        while len(self.min_heap) > k:
            heapq.heappop(self.min_heap)

    def add(self, val: int) -> int:
        heapq.heappush(self.min_heap, val)
        if len(self.min_heap) > self.k:
            heapq.heappop(self.min_heap)
        return self.min_heap[0]

Complexity: O(log k) Add | O(k) Space

#1046 LAST STONE WEIGHT

Smash stones together using a max-heap until one or zero remain.

import heapq

def lastStoneWeight(stones: list[int]) -> int:
    max_heap = [-s for s in stones]
    heapq.heapify(max_heap)

    while len(max_heap) > 1:
        s1 = heapq.heappop(max_heap)
        s2 = heapq.heappop(max_heap)
        if s1 != s2:
            heapq.heappush(max_heap, s1 - s2)
    
    return -max_heap[0] if max_heap else 0

Complexity: O(n log n) Time | O(n) Space

#973 K CLOSEST POINTS

Find the k closest points to (0, 0) on a 2D plane.

import heapq

def kClosest(points: list[list[int]], k: int) -> list[list[int]]:
    max_heap = []
    for x, y in points:
        dist = -(x**2 + y**2)
        heapq.heappush(max_heap, (dist, x, y))
        if len(max_heap) > k:
            heapq.heappop(max_heap)
    
    return [[x, y] for dist, x, y in max_heap]

Complexity: O(n log k) Time | O(k) Space

#621 TASK SCHEDULER

Minimum time to finish tasks with a cooling period.

from collections import Counter, deque
import heapq

def leastInterval(tasks: list[str], n: int) -> int:
    counts = Counter(tasks)
    max_heap = [-cnt for cnt in counts.values()]
    heapq.heapify(max_heap)
    
    time = 0
    wait_q = deque() # [-cnt, available_time]
    
    while max_heap or wait_q:
        time += 1
        if max_heap:
            cnt = 1 + heapq.heappop(max_heap)
            if cnt != 0:
                wait_q.append([cnt, time + n])
        
        if wait_q and wait_q[0][1] == time:
            heapq.heappush(max_heap, wait_q.popleft()[0])
            
    return time

Complexity: O(Total Time) | O(1) Space (max 26 tasks)

#355 DESIGN TWITTER

Simplified Twitter with feed, follow, and post functionality.

import heapq
from collections import defaultdict

class Twitter:
    def __init__(self):
        self.count = 0
        self.tweets = defaultdict(list) # userId: [(count, tweetId)]
        self.followMap = defaultdict(set) # userId: {followeeId}

    def postTweet(self, userId: int, tweetId: int):
        self.tweets[userId].append((self.count, tweetId))
        self.count -= 1

    def getNewsFeed(self, userId: int) -> list[int]:
        res, min_heap = [], []
        self.followMap[userId].add(userId)
        for followeeId in self.followMap[userId]:
            if followeeId in self.tweets:
                index = len(self.tweets[followeeId]) - 1
                count, tweetId = self.tweets[followeeId][index]
                min_heap.append((count, tweetId, followeeId, index - 1))
        heapq.heapify(min_heap)
        while min_heap and len(res) < 10:
            count, tweetId, followeeId, idx = heapq.heappop(min_heap)
            res.append(tweetId)
            if idx >= 0:
                cnt, tid = self.tweets[followeeId][idx]
                heapq.heappush(min_heap, (cnt, tid, followeeId, idx - 1))
        return res

    def follow(self, followerId: int, followeeId: int):
        self.followMap[followerId].add(followeeId)

    def unfollow(self, followerId: int, followeeId: int):
        if followeeId in self.followMap[followerId]:
            self.followMap[followerId].remove(followeeId)

Complexity: O(k log n) Feed | O(1) Post

#1882 PROCESS TASKS

Assign tasks to available servers with weights and indices.

import heapq

def assignTasks(servers: list[int], tasks: list[int]) -> list[int]:
    res = [0] * len(tasks)
    # available: (weight, index)
    available = [(servers[i], i) for i in range(len(servers))]
    heapq.heapify(available)
    # unavailable: (time_available, weight, index)
    unavailable = []
    
    t = 0
    for i in range(len(tasks)):
        t = max(t, i)
        if not available:
            t = unavailable[0][0]
        
        while unavailable and unavailable[0][0] <= t:
            time, w, idx = heapq.heappop(unavailable)
            heapq.heappush(available, (w, idx))
            
        w, idx = heapq.heappop(available)
        res[i] = idx
        heapq.heappush(unavailable, (t + tasks[i], w, idx))
    return res

Complexity: O(m log n) Time | O(n) Space

#1834 SINGLE-THREADED CPU

Process tasks on a CPU using processing time and indices.

import heapq

def getOrder(tasks: list[list[int]]) -> list[int]:
    for i, t in enumerate(tasks):
        t.append(i)
    tasks.sort(key=lambda x: x[0])
    
    res, min_heap = [], []
    i, time = 0, tasks[0][0]
    
    while min_heap or i < len(tasks):
        while i < len(tasks) and tasks[i][0] <= time:
            heapq.heappush(min_heap, (tasks[i][1], tasks[i][2]))
            i += 1
        
        if not min_heap:
            time = tasks[i][0]
        else:
            proc_time, index = heapq.heappop(min_heap)
            time += proc_time
            res.append(index)
    return res

Complexity: O(n log n) Time | O(n) Space

#295 MEDIAN DATA STREAM

Efficiently calculate the median of a growing stream of numbers.

import heapq

class MedianFinder:
    def __init__(self):
        # max_heap for small half, min_heap for large half
        self.small, self.large = [], []

    def addNum(self, num: int):
        heapq.heappush(self.small, -num)
        # Check order
        if self.small and self.large and (-self.small[0] > self.large[0]):
            val = -heapq.heappop(self.small)
            heapq.heappush(self.large, val)
        # Check size balance
        if len(self.small) > len(self.large) + 1:
            val = -heapq.heappop(self.small)
            heapq.heappush(self.large, val)
        if len(self.large) > len(self.small) + 1:
            val = heapq.heappop(self.large)
            heapq.heappush(self.small, -val)

    def findMedian(self) -> float:
        if len(self.small) > len(self.large):
            return -self.small[0]
        if len(self.large) > len(self.small):
            return self.large[0]
        return (-self.small[0] + self.large[0]) / 2.0

Complexity: O(log n) Add | O(1) Median

#23 MERGE K SORTED LISTS

Merge multiple sorted linked lists into one using a heap.

import heapq

def mergeKLists(lists: list[Optional[ListNode]]) -> Optional[ListNode]:
    min_heap = []
    for i, node in enumerate(lists):
        if node:
            heapq.heappush(min_heap, (node.val, i, node))
            
    dummy = ListNode()
    curr = dummy
    while min_heap:
        val, i, node = heapq.heappop(min_heap)
        curr.next = node
        curr = curr.next
        if node.next:
            heapq.heappush(min_heap, (node.next.val, i, node.next))
    return dummy.next

Complexity: O(n log k) Time | O(k) Space

#767 REORGANIZE STRING

Rearrange characters so that no two adjacent characters are the same.

import heapq
from collections import Counter

def reorganizeString(s: str) -> str:
    count = Counter(s)
    max_heap = [[-cnt, char] for char, cnt in count.items()]
    heapq.heapify(max_heap)
    
    prev = None
    res = ""
    while max_heap or prev:
        if prev and not max_heap:
            return ""
        
        cnt, char = heapq.heappop(max_heap)
        res += char
        cnt += 1
        
        if prev:
            heapq.heappush(max_heap, prev)
            prev = None
        
        if cnt != 0:
            prev = [cnt, char]
            
    return res

Complexity: O(n log 26) Time | O(26) Space

#692 TOP K FREQUENT WORDS

Find k most frequent words, sorted by frequency and lexicographically.

import heapq
from collections import Counter

def topKFrequentWords(words: list[str], k: int) -> list[str]:
    count = Counter(words)
    # Heap stores (-freq, word) for lexicographical order
    heap = [(-freq, word) for word, freq in count.items()]
    heapq.heapify(heap)
    return [heapq.heappop(heap)[1] for _ in range(k)]

Complexity: O(n + k log n) Time | O(n) Space

#1642 FURTHEST BUILDING

Reach the furthest building using bricks and ladders.

import heapq

def furthestBuilding(heights: list[int], bricks: int, ladders: int) -> int:
    heap = [] # Store brick usage to replace with ladders
    for i in range(len(heights) - 1):
        diff = heights[i+1] - heights[i]
        if diff > 0:
            heapq.heappush(heap, diff)
            if len(heap) > ladders:
                bricks -= heapq.heappop(heap)
            if bricks < 0:
                return i
    return len(heights) - 1

Complexity: O(n log l) Time | O(l) Space

#2336 SMALLEST IN INFINITE SET

Maintain an infinite set and pop the smallest element.

import heapq

class SmallestInfiniteSet:
    def __init__(self):
        self.added_back = []
        self.added_set = set()
        self.current_min = 1

    def popSmallest(self) -> int:
        if self.added_back:
            res = heapq.heappop(self.added_back)
            self.added_set.remove(res)
            return res
        self.current_min += 1
        return self.current_min - 1

    def addBack(self, num: int):
        if num < self.current_min and num not in self.added_set:
            heapq.heappush(self.added_back, num)
            self.added_set.add(num)

Complexity: O(log n) Pop/Add | O(n) Space
