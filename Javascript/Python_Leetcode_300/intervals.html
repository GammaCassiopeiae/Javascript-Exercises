#53 MAXIMUM SUBARRAY

Find the contiguous subarray with the largest sum (Kadane's Algorithm).

def maxSubArray(nums: list[int]) -> int:
    res = nums[0]
    total = 0
    for n in nums:
        total += n
        res = max(res, total)
        if total < 0:
            total = 0
    return res

Complexity: O(n) Time | O(1) Space

#55 JUMP GAME

Determine if you can reach the last index.

def canJump(nums: list[int]) -> bool:
    goal = len(nums) - 1
    for i in range(len(nums) - 2, -1, -1):
        if i + nums[i] >= goal:
            goal = i
    return goal == 0

Complexity: O(n) Time | O(1) Space

#45 JUMP GAME II

Find minimum number of jumps to reach the last index.

def jump(nums: list[int]) -> int:
    res = 0
    l = r = 0
    while r < len(nums) - 1:
        farthest = 0
        for i in range(l, r + 1):
            farthest = max(farthest, i + nums[i])
        l = r + 1
        r = farthest
        res += 1
    return res

Complexity: O(n) Time | O(1) Space

#134 GAS STATION

Find the starting gas station index to complete the circuit.

def canCompleteCircuit(gas: list[int], cost: list[int]) -> int:
    if sum(gas) < sum(cost): return -1
    total = 0
    res = 0
    for i in range(len(gas)):
        total += (gas[i] - cost[i])
        if total < 0:
            total = 0
            res = i + 1
    return res

Complexity: O(n) Time | O(1) Space

#846 HAND OF STRAIGHTS

Determine if cards can be rearranged into groups of consecutive cards.

def isNStraightHand(hand: list[int], groupSize: int) -> bool:
    if len(hand) % groupSize: return False
    count = collections.Counter(hand)
    minHeap = list(count.keys())
    heapq.heapify(minHeap)
    while minHeap:
        first = minHeap[0]
        for i in range(first, first + groupSize):
            if i not in count: return False
            count[i] -= 1
            if count[i] == 0:
                if i != minHeap[0]: return False
                heapq.heappop(minHeap)
    return True

Complexity: O(n log n) Time | O(n) Space

#56 MERGE INTERVALS

Merge all overlapping intervals.

def merge(intervals: list[list[int]]) -> list[list[int]]:
    intervals.sort(key=lambda i: i[0])
    output = [intervals[0]]
    for start, end in intervals[1:]:
        lastEnd = output[-1][1]
        if start <= lastEnd:
            output[-1][1] = max(lastEnd, end)
        else:
            output.append([start, end])
    return output

Complexity: O(n log n) Time | O(n) Space

#57 INSERT INTERVAL

Insert a new interval into a sorted list and merge if necessary.

def insert(intervals: list[list[int]], newInterval: list[int]) -> list[list[int]]:
    res = []
    for i in range(len(intervals)):
        if newInterval[1] < intervals[i][0]:
            res.append(newInterval)
            return res + intervals[i:]
        elif newInterval[0] > intervals[i][1]:
            res.append(intervals[i])
        else:
            newInterval = [min(newInterval[0], intervals[i][0]), max(newInterval[1], intervals[i][1])]
    res.append(newInterval)
    return res

Complexity: O(n) Time | O(n) Space

#435 NON-OVERLAPPING INTERVALS

Find the minimum number of intervals to remove to make the rest non-overlapping.

def eraseOverlapIntervals(intervals: list[list[int]]) -> int:
    intervals.sort()
    res = 0
    prevEnd = intervals[0][1]
    for start, end in intervals[1:]:
        if start >= prevEnd:
            prevEnd = end
        else:
            res += 1
            prevEnd = min(end, prevEnd)
    return res

Complexity: O(n log n) Time | O(1) Space

def canAttendMeetings(intervals: list[list[int]]) -> bool:

#253 MEETING ROOMS II

Find minimum number of conference rooms required.

def minMeetingRooms(intervals: list[list[int]]) -> int:
    start = sorted([i[0] for i in intervals])
    end = sorted([i[1] for i in intervals])
    res, count = 0, 0
    s, e = 0, 0
    while s < len(intervals):
        if start[s] < end[e]:
            s += 1; count += 1
        else:
            e += 1; count -= 1
        res = max(res, count)
    return res

Complexity: O(n log n) Time | O(n) Space

#763 PARTITION LABELS

Partition string into as many parts as possible such that each letter appears in at most one part.

def partitionLabels(s: str) -> list[int]:
    lastIndex = {c: i for i, c in enumerate(s)}
    res = []
    size, end = 0, 0
    for i, c in enumerate(s):
        size += 1
        end = max(end, lastIndex[c])
        if i == end:
            res.append(size)
            size = 0
    return res

Complexity: O(n) Time | O(1) Space

#678 VALID PARENTHESIS STRING

Check if string is valid given '(', ')' and '*'.

def checkValidString(s: str) -> bool:
    leftMin, leftMax = 0, 0
    for c in s:
        if c == "(":
            leftMin, leftMax = leftMin + 1, leftMax + 1
        elif c == ")":
            leftMin, leftMax = leftMin - 1, leftMax - 1
        else:
            leftMin, leftMax = leftMin - 1, leftMax + 1
        if leftMax < 0: return False
        if leftMin < 0: leftMin = 0
    return leftMin == 0

Complexity: O(n) Time | O(1) Space

#621 TASK SCHEDULER

Find the least number of units of time to finish all tasks.

def leastInterval(tasks: list[str], n: int) -> int:
    count = collections.Counter(tasks)
    maxHeap = [-cnt for cnt in count.values()]
    heapq.heapify(maxHeap)
    time = 0
    q = collections.deque() # [cnt, idleTime]
    while maxHeap or q:
        time += 1
        if maxHeap:
            cnt = 1 + heapq.heappop(maxHeap)
            if cnt: q.append([cnt, time + n])
        if q and q[0][1] == time:
            heapq.heappush(maxHeap, q.popleft()[0])
    return time

Complexity: O(n * log 26) Time | O(1) Space

#1899 MERGE TRIPLETS

Check if target triplet can be formed by merging other triplets.

def mergeTriplets(triplets: list[list[int]], target: list[int]) -> bool:
    good = set()
    for t in triplets:
        if t[0] > target[0] or t[1] > target[1] or t[2] > target[2]:
            continue
        for i, v in enumerate(t):
            if v == target[i]:
                good.add(i)
    return len(good) == 3

Complexity: O(n) Time | O(1) Space

#406 QUEUE RECONSTRUCTION

Reconstruct queue based on height and number of people in front.

def reconstructQueue(people: list[list[int]]) -> list[list[int]]:
    people.sort(key=lambda x: (-x[0], x[1]))
    res = []
    for p in people:
        res.insert(p[1], p)
    return res

Complexity: O(n²) Time | O(n) Space

#1024 VIDEO STITCHING

Find min clips to cover the time interval [0, T].

def videoStitching(clips: list[list[int]], time: int) -> int:
    res = 0
    curEnd, nextEnd = 0, 0
    clips.sort()
    i = 0
    while curEnd < time:
        while i < len(clips) and clips[i][0] <= curEnd:
            nextEnd = max(nextEnd, clips[i][1])
            i += 1
        if curEnd == nextEnd: return -1
        curEnd = nextEnd
        res += 1
    return res

Complexity: O(n log n) Time | O(1) Space

#135 CANDY

Find minimum candies to distribute to children based on ratings.

def candy(ratings: list[int]) -> int:
    res = [1] * len(ratings)
    for i in range(1, len(ratings)):
        if ratings[i] > ratings[i-1]:
            res[i] = res[i-1] + 1
    for i in range(len(ratings) - 2, -1, -1):
        if ratings[i] > ratings[i+1]:
            res[i] = max(res[i], res[i+1] + 1)
    return sum(res)

Complexity: O(n) Time | O(n) Space

#330 PATCHING ARRAY

Find min patches to make all sums from 1 to n possible.

def minPatches(nums: list[int], n: int) -> int:
    miss = 1
    res = 0
    i = 0
    while miss <= n:
        if i < len(nums) and nums[i] <= miss:
            miss += nums[i]
            i += 1
        else:
            miss += miss
            res += 1
    return res

Complexity: O(log n + len(nums)) Time | O(1) Space
