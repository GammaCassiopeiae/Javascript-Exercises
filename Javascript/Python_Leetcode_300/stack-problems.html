#20 VALID PARENTHESES

Determine if the input string has valid closing brackets in the correct order.

def isValid(s: str) -> bool:
    stack = []
    closeToOpen = {")": "(", "]": "[", "}": "{"}

    for c in s:
        if c in closeToOpen:
            if stack and stack[-1] == closeToOpen[c]:
                stack.pop()
            else:
                return False
        else:
            stack.append(c)
    return True if not stack else False

Complexity: O(n) Time | O(n) Space

#682 BASEBALL GAME

Keep track of scores using a stack based on a series of operations.

def calPoints(operations: list[str]) -> int:
    stack = []
    for op in operations:
        if op == "+":
            stack.append(stack[-1] + stack[-2])
        elif op == "D":
            stack.append(2 * stack[-1])
        elif op == "C":
            stack.pop()
        else:
            stack.append(int(op))
    return sum(stack)

Complexity: O(n) Time | O(n) Space

#225 IMPLEMENT STACK USING QUEUES

Implement a last-in-first-out (LIFO) stack using only two queues.

from collections import deque

class MyStack:
    def __init__(self):
        self.q = deque()

    def push(self, x: int) -> None:
        self.q.append(x)
        for _ in range(len(self.q) - 1):
            self.q.append(self.q.popleft())

    def pop(self) -> int:
        return self.q.popleft()

    def top(self) -> int:
        return self.q[0]

    def empty(self) -> bool:
        return len(self.q) == 0

Complexity: O(n) Push, O(1) Pop | O(n) Space

#155 MIN STACK

Design a stack that supports push, pop, top, and retrieving the minimum element in constant time.

class MinStack:
    def __init__(self):
        self.stack = []
        self.minStack = []

    def push(self, val: int) -> None:
        self.stack.append(val)
        val = min(val, self.minStack[-1] if self.minStack else val)
        self.minStack.append(val)

    def pop(self) -> None:
        self.stack.pop()
        self.minStack.pop()

    def top(self) -> int:
        return self.stack[-1]

    def getMin(self) -> int:
        return self.minStack[-1]

Complexity: O(1) Time | O(n) Space

#150 EVALUATE REVERSE POLISH NOTATION

Evaluate the value of an arithmetic expression in Reverse Polish Notation.

def evalRPN(tokens: list[str]) -> int:
    stack = []
    for c in tokens:
        if c == "+":
            stack.append(stack.pop() + stack.pop())
        elif c == "-":
            a, b = stack.pop(), stack.pop()
            stack.append(b - a)
        elif c == "*":
            stack.append(stack.pop() * stack.pop())
        elif c == "/":
            a, b = stack.pop(), stack.pop()
            stack.append(int(float(b) / a))
        else:
            stack.append(int(c))
    return stack[0]

Complexity: O(n) Time | O(n) Space

#22 GENERATE PARENTHESES

Generate all combinations of well-formed parentheses given n pairs.

def generateParenthesis(n: int) -> list[str]:
    stack = []
    res = []

    def backtrack(openN, closedN):
        if openN == closedN == n:
            res.append("".join(stack))
            return

        if openN < n:
            stack.append("(")
            backtrack(openN + 1, closedN)
            stack.pop()
        if closedN < openN:
            stack.append(")")
            backtrack(openN, closedN + 1)
            stack.pop()

    backtrack(0, 0)
    return res

Complexity: O(4ⁿ / √n) Time | O(n) Space

#739 DAILY TEMPERATURES

Find how many days you have to wait until a warmer temperature.

def dailyTemperatures(temperatures: list[int]) -> list[int]:
    res = [0] * len(temperatures)
    stack = []  # [temp, index]

    for i, t in enumerate(temperatures):
        while stack and t > stack[-1][0]:
            stackTemp, stackInd = stack.pop()
            res[stackInd] = i - stackInd
        stack.append((t, i))
    return res

Complexity: O(n) Time | O(n) Space

#853 CAR FLEET

Determine how many car fleets will arrive at the destination.

def carFleet(target: int, position: list[int], speed: list[int]) -> int:
    pair = [[p, s] for p, s in zip(position, speed)]
    stack = []
    for p, s in sorted(pair)[::-1]:  # Reverse Sorted
        stack.append((target - p) / s)
        if len(stack) >= 2 and stack[-1] <= stack[-2]:
            stack.pop()
    return len(stack)

Complexity: O(n log n) Time | O(n) Space

#901 ONLINE STOCK SPAN

Calculate the span of a stock's price today.

class StockSpanner:
    def __init__(self):
        self.stack = []  # pair: [price, span]

    def next(self, price: int) -> int:
        span = 1
        while self.stack and self.stack[-1][0] <= price:
            span += self.stack.pop()[1]
        self.stack.append([price, span])
        return span

Complexity: O(1) Amortized Time | O(n) Space

#71 SIMPLIFY PATH

Convert an absolute Unix-style path to its simplified canonical path.

def simplifyPath(path: str) -> str:
    stack = []
    for part in path.split("/"):
        if part == "..":
            if stack:
                stack.pop()
        elif part == "." or not part:
            continue
        else:
            stack.append(part)
    return "/" + "/".join(stack)

Complexity: O(n) Time | O(n) Space

#394 DECODE STRING

Decode an encoded string where k[string] means string is repeated k times.

def decodeString(s: str) -> str:
    stack = []
    for char in s:
        if char != "]":
            stack.append(char)
        else:
            substr = ""
            while stack[-1] != "[":
                substr = stack.pop() + substr
            stack.pop()  # remove [
            
            k = ""
            while stack and stack[-1].isdigit():
                k = stack.pop() + k
            stack.append(int(k) * substr)
    return "".join(stack)

Complexity: O(n) Time | O(n) Space

#735 ASTEROID COLLISION

Find the state of asteroids after all collisions.

def asteroidCollision(asteroids: list[int]) -> list[int]:
    stack = []
    for a in asteroids:
        while stack and a < 0 < stack[-1]:
            if stack[-1] < -a:
                stack.pop()
                continue
            elif stack[-1] == -a:
                stack.pop()
            break
        else:
            stack.append(a)
    return stack

Complexity: O(n) Time | O(n) Space

#2390 REMOVING STARS FROM A STRING

Remove the closest non-star character to the left of each star.

def removeStars(s: str) -> str:
    stack = []
    for c in s:
        if c == "*":
            stack.pop()
        else:
            stack.append(c)
    return "".join(stack)

Complexity: O(n) Time | O(n) Space

#456 132 PATTERN

Find if there is a 132 pattern in the array.

def find132pattern(nums: list[int]) -> bool:
    stack = []  # [num, minLeft]
    curMin = nums[0]

    for n in nums[1:]:
        while stack and n >= stack[-1][0]:
            stack.pop()
        if stack and n > stack[-1][1]:
            return True
        
        stack.append([n, curMin])
        curMin = min(curMin, n)
    return False

Complexity: O(n) Time | O(n) Space

#84 LARGEST RECTANGLE IN HISTOGRAM

Find the area of the largest rectangle in the histogram.

def largestRectangleArea(heights: list[int]) -> int:
    maxArea = 0
    stack = []  # (index, height)

    for i, h in enumerate(heights):
        start = i
        while stack and stack[-1][1] > h:
            index, height = stack.pop()
            maxArea = max(maxArea, height * (i - index))
            start = index
        stack.append((start, h))

    for i, h in stack:
        maxArea = max(maxArea, h * (len(heights) - i))
    return maxArea

Complexity: O(n) Time | O(n) Space
