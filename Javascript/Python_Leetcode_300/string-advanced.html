#208 IMPLEMENT TRIE

Implement a trie with insert, search, and startsWith methods.

class TrieNode:
    def __init__(self):
        self.children = {}
        self.endOfWord = False

class Trie:
    def __init__(self):
        self.root = TrieNode()
    def insert(self, word: str) -> None:
        cur = self.root
        for c in word:
            if c not in cur.children:
                cur.children[c] = TrieNode()
            cur = cur.children[c]
        cur.endOfWord = True
    def search(self, word: str) -> bool:
        cur = self.root
        for c in word:
            if c not in cur.children: return False
            cur = cur.children[c]
        return cur.endOfWord
    def startsWith(self, prefix: str) -> bool:
        cur = self.root
        for c in prefix:
            if c not in cur.children: return False
            cur = cur.children[c]
        return True

Complexity: O(n) per operation | O(n * m) Space

#211 DESIGN ADD/SEARCH WORDS

Support adding words and searching with '.' as a wildcard.

class WordDictionary:
    def __init__(self):
        self.root = TrieNode()
    def addWord(self, word: str) -> None:
        cur = self.root
        for c in word:
            if c not in cur.children:
                cur.children[c] = TrieNode()
            cur = cur.children[c]
        cur.endOfWord = True
    def search(self, word: str) -> bool:
        def dfs(j, root):
            cur = root
            for i in range(j, len(word)):
                c = word[i]
                if c == ".":
                    for child in cur.children.values():
                        if dfs(i + 1, child): return True
                    return False
                else:
                    if c not in cur.children: return False
                    cur = cur.children[c]
            return cur.endOfWord
        return dfs(0, self.root)

Complexity: O(n) Add, O(2^m) Search | O(n * m) Space

#212 WORD SEARCH II

Find all words from a dictionary in a 2D grid of characters.

def findWords(board: list[list[str]], words: list[str]) -> list[str]:
    root = TrieNode()
    for w in words: root.addWord(w)
    ROWS, COLS = len(board), len(board[0])
    res, visit = set(), set()
    def dfs(r, c, node, word):
        if (r < 0 or c < 0 or r == ROWS or c == COLS or
            (r, c) in visit or board[r][c] not in node.children):
            return
        visit.add((r, c))
        node = node.children[board[r][c]]
        word += board[r][c]
        if node.endOfWord: res.add(word)
        dfs(r + 1, c, node, word)
        dfs(r - 1, c, node, word)
        dfs(r, c + 1, node, word)
        dfs(r, c - 1, node, word)
        visit.remove((r, c))
    for r in range(ROWS):
        for c in range(COLS):
            dfs(r, c, root, "")
    return list(res)

Complexity: O(M * 4 * 3^(L-1)) Time | O(N * L) Space

#1392 LONGEST HAPPY PREFIX

Find the longest prefix that is also a suffix (excluding the string itself).

def longestPrefix(s: str) -> str:
    res = ""
    l, r = 0, 0
    mod = 10**9 + 7
    P = 31
    h_l, h_r = 0, 0
    p_pow = 1
    for i in range(len(s) - 1):
        h_l = (h_l * P + (ord(s[i]) - ord('a') + 1)) % mod
        h_r = (h_r + (ord(s[len(s) - 1 - i]) - ord('a') + 1) * p_pow) % mod
        p_pow = (p_pow * P) % mod
        if h_l == h_r:
            res = s[:i+1]
    return res

Complexity: O(n) Time | O(1) Space

#745 PREFIX AND SUFFIX SEARCH

Design a data structure that returns the index of a word with given prefix and suffix.

class WordFilter:
    def __init__(self, words: list[str]):
        self.trie = {}
        for idx, word in enumerate(words):
            long = word + "#" + word
            for i in range(len(word) + 1):
                cur = self.trie
                for j in range(i, len(long)):
                    char = long[j]
                    if char not in cur: cur[char] = {}
                    cur = cur[char]
                    cur["index"] = idx
    def f(self, prefix: str, suffix: str) -> int:
        cur = self.trie
        for char in suffix + "#" + prefix:
            if char not in cur: return -1
            cur = cur[char]
        return cur["index"]

Complexity: O(L²) Build, O(L) Query | O(L²) Space

#648 REPLACE WORDS

Replace words in a sentence with their shortest root from a dictionary.

def replaceWords(dictionary: list[str], sentence: str) -> str:
    root = TrieNode()
    for w in dictionary: root.addWord(w)
    res = []
    for word in sentence.split():
        cur = root
        prefix = ""
        for c in word:
            if c not in cur.children or cur.endOfWord: break
            prefix += c
            cur = cur.children[c]
        res.append(prefix if cur.endOfWord else word)
    return " ".join(res)

Complexity: O(N + M) Time | O(D) Space

#676 MAGIC DICTIONARY

Find if changing exactly one character can result in a word in the dictionary.

class MagicDictionary:
    def __init__(self):
        self.root = TrieNode()
    def buildDict(self, dictionary: list[str]) -> None:
        for w in dictionary: self.root.addWord(w)
    def search(self, searchWord: str) -> bool:
        def dfs(node, i, changed):
            if i == len(searchWord):
                return changed and node.endOfWord
            char = searchWord[i]
            for c in node.children:
                if c == char:
                    if dfs(node.children[c], i + 1, changed): return True
                elif not changed:
                    if dfs(node.children[c], i + 1, True): return True
            return False
        return dfs(self.root, 0, False)

Complexity: O(N * L) Build, O(L * 26) Search | O(N * L) Space

#677 MAP SUM PAIRS

Sum values of all keys that have a given prefix.

class MapSum:
    def __init__(self):
        self.root = {}
        self.map = {}
    def insert(self, key: str, val: int) -> None:
        delta = val - self.map.get(key, 0)
        self.map[key] = val
        cur = self.root
        for c in key:
            if c not in cur: cur[c] = {"sum": 0}
            cur = cur[c]
            cur["sum"] += delta
    def sum(self, prefix: str) -> int:
        cur = self.root
        for c in prefix:
            if c not in cur: return 0
            cur = cur[c]
        return cur["sum"]

Complexity: O(L) per operation | O(N * L) Space

#1268 SEARCH SUGGESTIONS

Return up to three products with shared prefix for each character typed.

def suggestedProducts(products: list[str], searchWord: str) -> list[list[str]]:
    products.sort()
    res = []
    l, r = 0, len(products) - 1
    for i in range(len(searchWord)):
        c = searchWord[i]
        while l <= r and (len(products[l]) <= i or products[l][i] != c):
            l += 1
        while l <= r and (len(products[r]) <= i or products[r][i] != c):
            r -= 1
        res.append([])
        remain = r - l + 1
        for j in range(min(3, remain)):
            res[-1].append(products[l + j])
    return res

Complexity: O(N log N + M) Time | O(M) Space

#421 MAX XOR

Find maximum result of XORing two numbers in an array.

def findMaximumXOR(nums: list[int]) -> int:
    res, mask = 0, 0
    for i in range(31, -1, -1):
        mask |= (1 << i)
        found = set()
        for n in nums: found.add(n & mask)
        start = res | (1 << i)
        for f in found:
            if start ^ f in found:
                res = start
                break
    return res

Complexity: O(n) Time | O(n) Space

#214 SHORTEST PALINDROME

Find the shortest palindrome by adding characters at the front.

def shortestPalindrome(s: str) -> str:
    if not s: return ""
    rev = s[::-1]
    for i in range(len(s)):
        if s.startswith(rev[i:]):
            return rev[:i] + s
    return ""

Complexity: O(n²) Time | O(n) Space

#28 FIND FIRST OCCURRENCE

Return index of first occurrence of needle in haystack (KMP Algorithm).

def strStr(haystack: str, needle: str) -> int:
    if needle == "": return 0
    lps = [0] * len(needle)
    prevLPS, i = 0, 1
    while i < len(needle):
        if needle[i] == needle[prevLPS]:
            lps[i] = prevLPS + 1
            prevLPS += 1; i += 1
        elif prevLPS == 0: i += 1
        else: prevLPS = lps[prevLPS - 1]
    i = j = 0
    while i < len(haystack):
        if haystack[i] == needle[j]:
            i += 1; j += 1
        elif j == 0: i += 1
        else: j = lps[j - 1]
        if j == len(needle): return i - len(needle)
    return -1

Complexity: O(n + m) Time | O(m) Space

#1032 STREAM OF CHARACTERS

Check if any suffix of characters typed so far forms a word in the dictionary.

class StreamChecker:
    def __init__(self, words: list[str]):
        self.trie = {}
        self.stream = []
        for w in words:
            cur = self.trie
            for c in reversed(w):
                if c not in cur: cur[c] = {}
                cur = cur[c]
            cur["#"] = True
    def query(self, letter: str) -> bool:
        self.stream.append(letter)
        cur = self.trie
        for i in range(len(self.stream) - 1, -1, -1):
            c = self.stream[i]
            if c not in cur: return False
            cur = cur[c]
            if "#" in cur: return True
        return False

Complexity: O(L) per query | O(N * L) Space

#187 REPEATED DNA SEQUENCES

Find all 10-letter-long sequences that occur more than once.

def findRepeatedDnaSequences(s: str) -> list[str]:
    seen, res = set(), set()
    for l in range(len(s) - 9):
        cur = s[l : l + 10]
        if cur in seen: res.add(cur)
        seen.add(cur)
    return list(res)

Complexity: O(n) Time | O(n) Space
