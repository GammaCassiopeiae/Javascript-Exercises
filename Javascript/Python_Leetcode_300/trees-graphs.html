#226 INVERT BINARY TREE

Invert a binary tree (mirror it).

def invertTree(root: Optional[TreeNode]) -> Optional[TreeNode]:
    if not root: return None
    root.left, root.right = root.right, root.left
    invertTree(root.left)
    invertTree(root.right)
    return root

Complexity: O(n) Time | O(h) Space

#104 MAX DEPTH OF BINARY TREE

Find the length of the longest path from root to leaf.

def maxDepth(root: Optional[TreeNode]) -> int:
    if not root: return 0
    return 1 + max(maxDepth(root.left), maxDepth(root.right))

Complexity: O(n) Time | O(h) Space

#543 DIAMETER OF BINARY TREE

Longest path between any two nodes in a tree.

def diameterOfBinaryTree(root: Optional[TreeNode]) -> int:
    res = 0
    def dfs(node):
        nonlocal res
        if not node: return 0
        left, right = dfs(node.left), dfs(node.right)
        res = max(res, left + right)
        return 1 + max(left, right)
    dfs(root)
    return res

Complexity: O(n) Time | O(h) Space

#110 BALANCED BINARY TREE

Check if height difference of subtrees is at most 1.

def isBalanced(root: Optional[TreeNode]) -> bool:
    def dfs(node):
        if not node: return [True, 0]
        left, right = dfs(node.left), dfs(node.right)
        balanced = left[0] and right[0] and abs(left[1] - right[1]) <= 1
        return [balanced, 1 + max(left[1], right[1])]
    return dfs(root)[0]

Complexity: O(n) Time | O(h) Space

#100 SAME TREE

Check if two binary trees are structurally identical.

def isSameTree(p: Optional[TreeNode], q: Optional[TreeNode]) -> bool:
    if not p and not q: return True
    if not p or not q or p.val != q.val: return False
    return isSameTree(p.left, q.left) and isSameTree(p.right, q.right)

Complexity: O(n) Time | O(h) Space

#572 SUBTREE OF ANOTHER TREE

Check if a tree contains another tree as a subtree.

def isSubtree(root: Optional[TreeNode], subRoot: Optional[TreeNode]) -> bool:
    if not subRoot: return True
    if not root: return False
    if isSameTree(root, subRoot): return True
    return isSubtree(root.left, subRoot) or isSubtree(root.right, subRoot)

def isSameTree(p, q):
    if not p and not q: return True
    if p and q and p.val == q.val:
        return isSameTree(p.left, q.left) and isSameTree(p.right, q.right)
    return False

Complexity: O(m*n) Time | O(h) Space

#235 LCA OF A BST

Find LCA in a Binary Search Tree.

def lowestCommonAncestor(root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
    curr = root
    while curr:
        if p.val > curr.val and q.val > curr.val:
            curr = curr.right
        elif p.val < curr.val and q.val < curr.val:
            curr = curr.left
        else:
            return curr

Complexity: O(h) Time | O(1) Space

#112 PATH SUM

Root-to-leaf path sum equals target.

def hasPathSum(root: Optional[TreeNode], targetSum: int) -> bool:
    if not root: return False
    if not root.left and not root.right:
        return targetSum == root.val
    return (hasPathSum(root.left, targetSum - root.val) or 
            hasPathSum(root.right, targetSum - root.val))

Complexity: O(n) Time | O(h) Space

#145 POSTORDER TRAVERSAL

L-R-Root traversal (Iterative).

def postorderTraversal(root: Optional[TreeNode]) -> list[int]:
    if not root: return []
    res, stack = [], [root]
    while stack:
        node = stack.pop()
        res.append(node.val)
        if node.left: stack.append(node.left)
        if node.right: stack.append(node.right)
    return res[::-1]

Complexity: O(n) Time | O(h) Space

#783 MIN DISTANCE IN BST

Find the minimum difference between any two nodes.

def minDiffInBST(root: Optional[TreeNode]) -> int:
    prev, res = None, float('inf')
    def dfs(node):
        nonlocal prev, res
        if not node: return
        dfs(node.left)
        if prev:
            res = min(res, node.val - prev.val)
        prev = node
        dfs(node.right)
    dfs(root)
    return res

Complexity: O(n) Time | O(h) Space

#102 LEVEL ORDER TRAVERSAL

BFS level-by-level traversal.

from collections import deque
def levelOrder(root: Optional[TreeNode]) -> list[list[int]]:
    res, q = [], deque([root] if root else [])
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res

Complexity: O(n) Time | O(n) Space

#199 RIGHT SIDE VIEW

Nodes visible from the right side.

def rightSideView(root: Optional[TreeNode]) -> list[int]:
    res, q = [], deque([root] if root else [])
    while q:
        rightNode = None
        for _ in range(len(q)):
            node = q.popleft()
            rightNode = node
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        if rightNode: res.append(rightNode.val)
    return res

Complexity: O(n) Time | O(n) Space

#1448 COUNT GOOD NODES

Nodes with val >= max val on path from root.

def goodNodes(root: TreeNode) -> int:
    def dfs(node, maxVal):
        if not node: return 0
        res = 1 if node.val >= maxVal else 0
        maxVal = max(maxVal, node.val)
        res += dfs(node.left, maxVal)
        res += dfs(node.right, maxVal)
        return res
    return dfs(root, root.val)

Complexity: O(n) Time | O(h) Space

#98 VALIDATE BST

Check if a tree is a valid BST.

def isValidBST(root: Optional[TreeNode]) -> bool:
    def valid(node, left, right):
        if not node: return True
        if not (left < node.val < right): return False
        return (valid(node.left, left, node.val) and 
                valid(node.right, node.val, right))
    return valid(root, float("-inf"), float("inf"))

Complexity: O(n) Time | O(h) Space

#230 KTH SMALLEST IN BST

Find the kth smallest value using inorder traversal.

def kthSmallest(root: Optional[TreeNode], k: int) -> int:
    stack, curr = [], root
    while stack or curr:
        while curr:
            stack.append(curr)
            curr = curr.left
        curr = stack.pop()
        k -= 1
        if k == 0: return curr.val
        curr = curr.right

Complexity: O(h + k) Time | O(h) Space

#105 BUILD FROM TRAVERSALS

Reconstruct tree from preorder and inorder arrays.

def buildTree(preorder: list[int], inorder: list[int]) -> Optional[TreeNode]:
    if not preorder or not inorder: return None
    root = TreeNode(preorder[0])
    mid = inorder.index(preorder[0])
    root.left = buildTree(preorder[1:mid+1], inorder[:mid])
    root.right = buildTree(preorder[mid+1:], inorder[mid+1:])
    return root

Complexity: O(n) Time | O(n) Space

#103 ZIGZAG LEVEL ORDER

Level order traversal with alternating directions.

def zigzagLevelOrder(root: Optional[TreeNode]) -> list[list[int]]:
    res, q = [], deque([root] if root else [])
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        if len(res) % 2 == 1: level.reverse()
        res.append(level)
    return res

Complexity: O(n) Time | O(n) Space

#116 POPULATE NEXT POINTERS

Link all nodes at the same level.

def connect(root: 'Optional[Node]') -> 'Optional[Node]':
    curr, nxt = root, root.left if root else None
    while curr and nxt:
        curr.left.next = curr.right
        if curr.next:
            curr.right.next = curr.next.left
        curr = curr.next
        if not curr:
            curr, nxt = nxt, nxt.left
    return root

Complexity: O(n) Time | O(1) Space

#114 FLATTEN BINARY TREE

Flatten tree to a "linked list" in-place.

def flatten(root: Optional[TreeNode]) -> None:
    curr = root
    while curr:
        if curr.left:
            last = curr.left
            while last.right: last = last.right
            last.right = curr.right
            curr.right = curr.left
            curr.left = None
        curr = curr.right

Complexity: O(n) Time | O(1) Space

#173 BST ITERATOR

Inorder traversal iterator using a stack.

class BSTIterator:
    def __init__(self, root: Optional[TreeNode]):
        self.stack = []
        self._push_left(root)
    def _push_left(self, node):
        while node:
            self.stack.append(node)
            node = node.left
    def next(self) -> int:
        node = self.stack.pop()
        self._push_left(node.right)
        return node.val
    def hasNext(self) -> bool:
        return len(self.stack) > 0

Complexity: O(1) avg next() | O(h) Space

#669 TRIM A BST

Remove nodes outside [low, high] range.

def trimBST(root: Optional[TreeNode], low: int, high: int) -> Optional[TreeNode]:
    if not root: return None
    if root.val < low: return trimBST(root.right, low, high)
    if root.val > high: return trimBST(root.left, low, high)
    root.left = trimBST(root.left, low, high)
    root.right = trimBST(root.right, low, high)
    return root

Complexity: O(n) Time | O(h) Space

#337 HOUSE ROBBER III

Max amount you can rob without robbing adjacent nodes.

def rob(root: Optional[TreeNode]) -> int:
    def dfs(node):
        if not node: return [0, 0] # [robNode, skipNode]
        left, right = dfs(node.left), dfs(node.right)
        robNode = node.val + left[1] + right[1]
        skipNode = max(left) + max(right)
        return [robNode, skipNode]
    return max(dfs(root))

Complexity: O(n) Time | O(h) Space

#124 MAX PATH SUM

Find the maximum path sum between any two nodes.

def maxPathSum(root: Optional[TreeNode]) -> int:
    res = root.val
    def dfs(node):
        nonlocal res
        if not node: return 0
        leftMax = max(dfs(node.left), 0)
        rightMax = max(dfs(node.right), 0)
        res = max(res, node.val + leftMax + rightMax)
        return node.val + max(leftMax, rightMax)
    dfs(root)
    return res

Complexity: O(n) Time | O(h) Space

#297 SERIALIZE TREE

Convert tree to string and back.

class Codec:
    def serialize(self, root):
        res = []
        def dfs(node):
            if not node: res.append("N"); return
            res.append(str(node.val))
            dfs(node.left)
            dfs(node.right)
        dfs(root)
        return ",".join(res)
    def deserialize(self, data):
        vals = data.split(",")
        self.i = 0
        def dfs():
            if vals[self.i] == "N": self.i += 1; return None
            node = TreeNode(int(vals[self.i])); self.i += 1
            node.left, node.right = dfs(), dfs()
            return node
        return dfs()

Complexity: O(n) Time | O(n) Space

#94 INORDER TRAVERSAL

Left-Root-Right traversal (Iterative).

def inorderTraversal(root: Optional[TreeNode]) -> list[int]:
    res, stack, curr = [], [], root
    while curr or stack:
        while curr:
            stack.append(curr); curr = curr.left
        curr = stack.pop()
        res.append(curr.val); curr = curr.right
    return res

Complexity: O(n) Time | O(h) Space

#144 PREORDER TRAVERSAL

Root-Left-Right traversal (Iterative).

def preorderTraversal(root: Optional[TreeNode]) -> list[int]:
    if not root: return []
    res, stack = [], [root]
    while stack:
        node = stack.pop()
        res.append(node.val)
        if node.right: stack.append(node.right)
        if node.left: stack.append(node.left)
    return res

Complexity: O(n) Time | O(h) Space

#101 SYMMETRIC TREE

Check if a tree is a mirror of itself.

def isSymmetric(root: Optional[TreeNode]) -> bool:
    def isMirror(t1, t2):
        if not t1 and not t2: return True
        if not t1 or not t2: return False
        return (t1.val == t2.val and isMirror(t1.left, t2.right) 
                and isMirror(t1.right, t2.left))
    return isMirror(root, root)

Complexity: O(n) Time | O(h) Space

#563 BINARY TREE TILT

Sum of absolute differences between subtree sums.

def findTilt(root: Optional[TreeNode]) -> int:
    tilt = 0
    def dfs(node):
        nonlocal tilt
        if not node: return 0
        left, right = dfs(node.left), dfs(node.right)
        tilt += abs(left - right)
        return node.val + left + right
    dfs(root)
    return tilt

Complexity: O(n) Time | O(h) Space

#404 SUM OF LEFT LEAVES

Sum of all leaves that are left children.

def sumOfLeftLeaves(root: Optional[TreeNode]) -> int:
    if not root: return 0
    res = 0
    if root.left and not root.left.left and not root.left.right:
        res += root.left.val
    return res + sumOfLeftLeaves(root.left) + sumOfLeftLeaves(root.right)

Complexity: O(n) Time | O(h) Space

#113 PATH SUM II

All root-to-leaf paths that sum to target.

def pathSum(root: Optional[TreeNode], targetSum: int) -> list[list[int]]:
    res = []
    def dfs(node, currSum, path):
        if not node: return
        currSum += node.val
        if not node.left and not node.right and currSum == targetSum:
            res.append(path + [node.val])
        dfs(node.left, currSum, path + [node.val])
        dfs(node.right, currSum, path + [node.val])
    dfs(root, 0, [])
    return res

Complexity: O(n^2) Time | O(h) Space

#257 BINARY TREE PATHS

All root-to-leaf paths in string format.

def binaryTreePaths(root: Optional[TreeNode]) -> list[str]:
    res = []
    def dfs(node, path):
        if not node: return
        path += str(node.val)
        if not node.left and not node.right:
            res.append(path)
        else:
            dfs(node.left, path + "->")
            dfs(node.right, path + "->")
    dfs(root, "")
    return res

Complexity: O(n) Time | O(h) Space

#108 ARRAY TO BST

Build a balanced BST from a sorted array.

def sortedArrayToBST(nums: list[int]) -> Optional[TreeNode]:
    if not nums: return None
    mid = len(nums) // 2
    root = TreeNode(nums[mid])
    root.left = sortedArrayToBST(nums[:mid])
    root.right = sortedArrayToBST(nums[mid+1:])
    return root

Complexity: O(n) Time | O(log n) Space

#701 INSERT INTO BST

Insert a value while maintaining BST properties.

def insertIntoBST(root: Optional[TreeNode], val: int) -> Optional[TreeNode]:
    if not root: return TreeNode(val)
    if val < root.val:
        root.left = insertIntoBST(root.left, val)
    else:

#450 DELETE NODE IN BST

Remove a node from BST and reorganize.

def deleteNode(root: Optional[TreeNode], key: int) -> Optional[TreeNode]:
    if not root: return None
    if key < root.val:
        root.left = deleteNode(root.left, key)
    elif key > root.val:
        root.right = deleteNode(root.right, key)
    else:
        if not root.left: return root.right
        if not root.right: return root.left
        curr = root.right
        while curr.left: curr = curr.left
        root.val = curr.val
        root.right = deleteNode(root.right, curr.val)
    return root

Complexity: O(h) Time | O(h) Space

#863 NODES DISTANCE K

Find all nodes at distance k from target node.

def distanceK(root: TreeNode, target: TreeNode, k: int) -> list[int]:
    adj = collections.defaultdict(list)
    def build(node, parent):
        if node and parent:
            adj[node.val].append(parent.val)
            adj[parent.val].append(node.val)
        if node.left: build(node.left, node)
        if node.right: build(node.right, node)
    build(root, None)
    res, q, visit = [], deque([(target.val, 0)]), {target.val}
    while q:
        val, d = q.popleft()
        if d == k: res.append(val)
        if d < k:
            for n in adj[val]:
                if n not in visit:
                    visit.add(n); q.append((n, d + 1))
    return res

Complexity: O(n) Time | O(n) Space

#107 LEVEL ORDER II

Bottom-up level order traversal.

def levelOrderBottom(root: Optional[TreeNode]) -> list[list[int]]:
    res, q = [], deque([root] if root else [])
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res[::-1]

Complexity: O(n) Time | O(n) Space

#117 POPULATE NEXT II

Same as #116 but for any binary tree.

def connect(root: 'Node') -> 'Node':
    if not root: return None
    q = deque([root])
    while q:
        prev = None
        for _ in range(len(q)):
            node = q.popleft()
            if prev: prev.next = node
            prev = node
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
    return root

Complexity: O(n) Time | O(n) Space

#236 LCA OF BINARY TREE

Find LCA in any binary tree.

def lowestCommonAncestor(root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
    if not root or root == p or root == q: return root
    left = lowestCommonAncestor(root.left, p, q)
    right = lowestCommonAncestor(root.right, p, q)
    if left and right: return root
    return left if left else right

Complexity: O(n) Time | O(h) Space
