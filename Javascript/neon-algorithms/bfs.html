function bfs(start) {
    let queue = [start];
    visited.add(start);
    while (queue.length > 0) {
        let node = queue.shift();
        for (let neighbor of adj[node]) {
            if (!visited.has(neighbor)) {
                visited.add(neighbor);
                queue.push(neighbor);
            }
        }
    }
}
