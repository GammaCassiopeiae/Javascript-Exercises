async function countingSort(arr) {
    let max = Math.max(...arr);
    let count = new Array(max + 1).fill(0);
    for (let x of arr) count[x]++;
    let k = 0;
    for (let i = 0; i <= max; i++) {
        while (count[i]-- > 0) {
            arr[k++] = i;
            render();
        }
    }
}
