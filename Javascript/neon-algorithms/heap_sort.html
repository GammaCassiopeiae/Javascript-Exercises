async function heapSort(arr) {
    let n = arr.length;
    for (let i = Math.floor(n / 2) - 1; i >= 0; i--)
        await heapify(arr, n, i);
    for (let i = n - 1; i > 0; i--) {
        await swap(0, i);
        await heapify(arr, i, 0);
    }
}
