async function insertionSort(arr) {
    for (let i = 1; i < arr.length; i++) {
        let key = arr[i];
        let j = i - 1;
        while (j >= 0 && arr[j] > key) {
            await highlight([j, j + 1]);
            arr[j + 1] = arr[j];
            j = j - 1;
            render();
        }
        arr[j + 1] = key;
    }
}
