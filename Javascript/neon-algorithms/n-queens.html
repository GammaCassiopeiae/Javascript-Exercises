function solve(row) {
    if (row == n) return true;
    for (let col = 0; col < n; col++) {
        if (isSafe(row, col)) {
            board[row][col] = 1;
            if (solve(row + 1)) return true;
            board[row][col] = 0;
        }
    }
}
