function permute(nums, start = 0) {
    if (start == nums.length) res.push([...nums]);
    for (let i = start; i < nums.length; i++) {
        [nums[start], nums[i]] = [nums[i], nums[start]];
        permute(nums, start + 1);
        [nums[start], nums[i]] = [nums[i], nums[start]];
    }
}
