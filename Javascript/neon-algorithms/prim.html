function prim(start) {
    dist[start] = 0;
    while (pq.isNotEmpty()) {
        let u = pq.pop();
        visited[u] = true;
        for (let {v, w} of adj[u]) {
            if (!visited[v] && w < dist[v]) {
                dist[v] = w;
                parent[v] = u;
                pq.push(v, dist[v]);
            }
        }
    }
}
