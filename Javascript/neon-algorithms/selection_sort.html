async function selectionSort(arr) {
    for (let i = 0; i < arr.length; i++) {
        let minIdx = i;
        for (let j = i + 1; j < arr.length; j++) {
            await highlight([minIdx, j]);
            if (arr[j] < arr[minIdx]) minIdx = j;
        }
        await swap(i, minIdx);
    }
}
