function solve(r, c) {
    if (r == 9) return true;
    if (c == 9) return solve(r+1, 0);
    if (grid[r][c] != 0) return solve(r, c+1);
    for (let n = 1; n <= 9; n++) {
        if (isValid(r, c, n)) {
            grid[r][c] = n;
            if (solve(r, c+1)) return true;
            grid[r][c] = 0;
        }
    }
}
