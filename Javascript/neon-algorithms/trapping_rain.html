function trap(height) {
    let l = 0, r = height.length - 1;
    let lMax = 0, rMax = 0, res = 0;
    while (l < r) {
        if (height[l] < height[r]) {
            height[l] >= lMax ? (lMax = height[l]) : (res += lMax - height[l]);
            l++;
        } else {
            height[r] >= rMax ? (rMax = height[r]) : (res += rMax - height[r]);
            r--;
        }
    }
}
